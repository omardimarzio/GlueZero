    ```typescript
    import { vi } from 'vitest'
    import { http, HttpResponse, type RequestHandler } from 'msw'
    import type { CanonicalSchema } from '@gluezero/mapper'
    import { createRouterBroker, type RouterBroker } from '../public-factory'
    import type { RouteDefinition } from '../types/route-definition'
    import type { GatewayConfig } from '@gluezero/gateway/http'
    import { server } from './msw-server'

    export interface RouterHarnessOptions {
      readonly debug?: boolean
      readonly schemas?: readonly CanonicalSchema[]
      readonly transforms?: Readonly<Record<string, (...args: unknown[]) => unknown>>
      readonly routes?: readonly RouteDefinition[]
      readonly gateway?: GatewayConfig
    }

    export interface RouterHarness {
      readonly broker: RouterBroker
      readonly steps: Array<{ step: string; snapshot: unknown }>
      reset(): void
      defineRoute(def: RouteDefinition): void
      mockServer(handlers: RequestHandler[]): void
      expectFetched(matcher: string | RegExp): number
      expectRetryAttempts(routeId: string): number
      collectedEvents: Array<{ topic: string; payload: unknown }>
    }

    export function createRouterHarness(options: RouterHarnessOptions = {}): RouterHarness {
      const steps: RouterHarness['steps'] = []
      const collectedEvents: RouterHarness['collectedEvents'] = []
      const tap = { onPipelineStep: (step: string, snapshot: unknown) => { steps.push({ step, snapshot }) } }

      const fetchTracker = { calls: [] as Array<{ url: string; method: string }> }
      // monkey-patch msw to track fetched calls
      const originalUse = server.use.bind(server)
      server.events.on('request:start', ({ request }) => {
        fetchTracker.calls.push({ url: request.url, method: request.method })
      })

      const broker = createRouterBroker({
        runtime: { tap, logLevel: 'silent' },
        ...(options.schemas && { canonicalModel: { schemas: [...options.schemas] } }),
        ...(options.transforms && { transforms: { ...options.transforms } }),
        ...(options.routes && { routes: [...options.routes] }),
        ...(options.gateway && { gateway: options.gateway }),
      })

      // Spy su publish via subscribe wildcard. BLOCKER 3 revision fix Plan 03-12:
      // RouterBroker espone `subscribe(...)` come delegate esplicito a MapperBroker.subscribe
      // (PLAN 03-12 Task 2). Niente più optional chaining `?.`.
      broker.subscribe('*' as unknown as string, (e: { topic: string; payload: unknown }) => {
        collectedEvents.push({ topic: e.topic, payload: e.payload })
      })

      return {
        broker,
        steps,
        collectedEvents,
        reset() {
          steps.length = 0
          collectedEvents.length = 0
          fetchTracker.calls = []
        },
        defineRoute(def) {
          broker.registerRoute(def)
        },
        mockServer(handlers) {
          server.use(...handlers)
        },
        expectFetched(matcher) {
          return fetchTracker.calls.filter((c) => typeof matcher === 'string' ? c.url.includes(matcher) : matcher.test(c.url)).length
        },
        expectRetryAttempts(routeId) {
          // Counter via tap snapshot route-executed events filtered by routeId
          return steps.filter((s) => s.step === 'event.route.executed' && JSON.stringify(s.snapshot).includes(routeId)).length
        },
      }
    }
    ```
