    // auth-strategy.ts — BearerHookAuth (D-72, SEC-01/SEC-02/ROUTE-07).
    //
    // Wrapper su gateway.auth.getToken/refresh consumer-provided con single-flight refresh
    // (Pattern 5 RESEARCH, Pitfall 5 fix): N caller concurrent ricevono UNA SOLA refresh invocation.

    import { createBrokerError } from '@sembridge/core'
    import type { AuthStrategy } from '../types/http-strategies'
    import type { AuthStrategyConfig } from '../types/gateway-config'

    export interface AuthStrategyOptions {
      readonly config: AuthStrategyConfig
    }

    export function createAuthStrategy(options: AuthStrategyOptions): AuthStrategy {
      const { config } = options
      const tokenCacheMs = config.tokenCacheMs ?? 0
      let cachedToken: string | undefined
      let cachedAt = 0
      let inflightRefresh: Promise<string> | null = null

      return {
        async getToken() {
          if (tokenCacheMs > 0 && cachedToken !== undefined && Date.now() - cachedAt < tokenCacheMs) {
            return cachedToken
          }
          const token = await config.getToken()
          if (token !== undefined && tokenCacheMs > 0) {
            cachedToken = token
            cachedAt = Date.now()
          }
          return token
        },
        ...(config.refresh && {
          async refresh() {
            if (!config.refresh) {
              throw createBrokerError({
                code: 'auth.refresh.unavailable',
                category: 'config', // BLOCKER 1 fix: ErrorCategory union NON include 'auth'; usiamo 'config' coerente con 03-08
                message: 'config.auth.refresh is not configured',
              })
            }
            // Single-flight (Pattern 5)
            if (inflightRefresh !== null) return inflightRefresh
            inflightRefresh = (async () => {
              try {
                const newToken = await config.refresh!()
                cachedToken = newToken
                cachedAt = Date.now()
                return newToken
              } finally {
                inflightRefresh = null
              }
            })()
            return inflightRefresh
          },
        }),
        isInflightRefresh() {
          return inflightRefresh !== null
        },
      }
    }
    ```

    // circuit-breaker.ts — PerRouteCircuitBreaker (D-99, opt-in DISABLED default).
    //
    // State machine: closed → open → half-open → closed.
    // - closed: requests pass; failure increments counter; success resets counter.
    // - open: fail-fast (canExecute=false); transition automatica a half-open dopo cooldownMs.
    // - half-open: 1 prova ammessa; success → closed; failure → open ricaricato.
    //
    // Default DISABLED: createCircuitBreakerStrategy() senza config = pass-through (canExecute=true).
    // Sliding window stats → V1.x.

    import type { CircuitBreakerStrategy } from '../types/http-strategies'
    import type { CircuitBreakerConfig } from '../types/gateway-config'

    interface CircuitState {
      state: 'closed' | 'open' | 'half-open'
      consecutiveFailures: number
      openedAt: number
    }

    export interface CircuitBreakerStrategyOptions {
      readonly config?: CircuitBreakerConfig
    }

    export function createCircuitBreakerStrategy(options: CircuitBreakerStrategyOptions = {}): CircuitBreakerStrategy {
      const config = options.config
      const states = new Map<string, CircuitState>()

      function getState(routeId: string): CircuitState {
        let s = states.get(routeId)
        if (!s) {
          s = { state: 'closed', consecutiveFailures: 0, openedAt: 0 }
          states.set(routeId, s)
        }
        return s
      }

      function maybeTransitionToHalfOpen(s: CircuitState): void {
        if (s.state === 'open' && config && Date.now() - s.openedAt >= config.cooldownMs) {
          s.state = 'half-open'
        }
      }

      return {
        canExecute(routeId) {
          if (!config) return true  // DISABLED default
          const s = getState(routeId)
          maybeTransitionToHalfOpen(s)
          return s.state !== 'open'
        },
        recordSuccess(routeId) {
          if (!config) return
          const s = getState(routeId)
          s.consecutiveFailures = 0
          if (s.state === 'half-open' || s.state === 'open') {
            s.state = 'closed'
            s.openedAt = 0
          }
        },
        recordFailure(routeId) {
          if (!config) return
          const s = getState(routeId)
          s.consecutiveFailures++
          if (s.state === 'half-open') {
            s.state = 'open'
            s.openedAt = Date.now()
            return
          }
          if (s.state === 'closed' && s.consecutiveFailures >= config.threshold) {
            s.state = 'open'
            s.openedAt = Date.now()
          }
        },
        getState(routeId) {
          if (!config) return 'closed'
          const s = getState(routeId)
          maybeTransitionToHalfOpen(s)
          return s.state
        },
      }
    }
    ```
