    // dedupe-strategy.ts — KeyBasedDedupe (D-74, ROUTE-11).
    //
    // Promise singleton per key: N caller concurrent con stessa key ricevono la STESSA Promise.
    // Pattern: Map<key, Promise>. Entry rilasciata in finally — sia su success che failure.
    //
    // Use case: GET /api/weather?city=Roma con dedupeKey 'weather:Roma' chiamato 5 volte
    // contemporaneamente → 1 sola fetch HTTP, 5 caller ricevono stesso payload.

    import type { DedupeStrategy } from '../types/http-strategies'

    export interface DedupeStrategyOptions {
      /** Cap entries inflight (per evitare DoS — default 1000). */
      readonly maxInflight?: number
    }

    export function createDedupeStrategy(options: DedupeStrategyOptions = {}): DedupeStrategy {
      const maxInflight = options.maxInflight ?? 1000
      const inflight = new Map<string, Promise<unknown>>()

      return {
        async execute<T>(key: string, fn: () => Promise<T>): Promise<T> {
          const existing = inflight.get(key)
          if (existing !== undefined) {
            return existing as Promise<T>
          }
          if (inflight.size >= maxInflight) {
            // DoS guard — esegue diretto senza dedup (warning sostenuto in dev mode plan 03-12)
            return await fn()
          }
          const promise = (async () => {
            try {
              return await fn()
            } finally {
              inflight.delete(key)
            }
          })()
          inflight.set(key, promise)
          return promise as Promise<T>
        },
        size() {
          return inflight.size
        },
        clear() {
          inflight.clear()
        },
      }
    }
    ```

    // backpressure-strategy.ts — Backpressure policy 6 types con priority-aware bypass (D-75, ROUTE-10, Pitfall 4 fix).
    //
    // Policy supportate (D-75):
    // - 'queue-bounded' { max: N } — accoda fino a N, drop-new su overflow (default) o drop-oldest (opt-in)
    // - 'drop' — drop fast: solo 1 task in volo, secondi rejected
    // - 'throttle' { perSec: N } — rate-limit: max N task/sec
    // - 'debounce' { waitMs: N } — esegue solo dopo N ms quiet (consecutive collapse)
    // - 'latest-only' — abort precedente, esegui nuovo
    // - 'merge'/'coalesce' — accumula task pendenti in uno (V1 minimal: alias 'latest-only' con flag)
    //
    // CRITICAL bypass: priority='critical' (es. system.error) BYPASSA tutte le policy
    // — chiusura Pitfall 4 (PRD §22.3 critical events sempre delivered).

    import { createBrokerError } from '@sembridge/core'
    import type { BackpressureStrategy } from '../types/http-strategies'
    import type { BackpressurePolicyConfig } from '@sembridge/routing'

    type RouteState = {
      readonly policy: BackpressurePolicyConfig
      pending: Array<{ task: () => Promise<unknown>; resolve: (v: unknown) => void; reject: (e: unknown) => void; controller: AbortController }>
      inFlight: number
      lastExecuteAt: number
      throttleCount: number
      throttleWindowStart: number
      debounceTimer: ReturnType<typeof setTimeout> | null
    }

    export interface BackpressureStrategyOptions {
      readonly defaultPolicy?: BackpressurePolicyConfig
      readonly resolvePolicy?: (routeId: string) => BackpressurePolicyConfig | undefined
      readonly dropOldest?: boolean
    }

    export function createBackpressureStrategy(options: BackpressureStrategyOptions = {}): BackpressureStrategy {
      const states = new Map<string, RouteState>()
      const dropOldest = options.dropOldest ?? false

      function getState(routeId: string): RouteState {
        let s = states.get(routeId)
        if (!s) {
          const policy = options.resolvePolicy?.(routeId) ?? options.defaultPolicy ?? { type: 'queue-bounded', max: 100 }
          s = { policy, pending: [], inFlight: 0, lastExecuteAt: 0, throttleCount: 0, throttleWindowStart: 0, debounceTimer: null }
          states.set(routeId, s)
        }
        return s
      }

      async function executeNow<T>(state: RouteState, task: () => Promise<T>): Promise<T> {
        state.inFlight++
        try {
          return await task()
        } finally {
          state.inFlight--
        }
      }

      return {
        async schedule<T>(routeId: string, priority: 'critical'|'high'|'normal'|'low', task: () => Promise<T>): Promise<T> {
          // CRITICAL bypass — Pitfall 4 fix
          if (priority === 'critical') {
            return await task()
          }
          const state = getState(routeId)

          switch (state.policy.type) {
            case 'queue-bounded': {
              const max = state.policy.max
              if (state.pending.length >= max) {
                if (dropOldest && state.pending.length > 0) {
                  const oldest = state.pending.shift()!
                  oldest.controller.abort('backpressure.dropped-oldest')
                  oldest.reject(createBrokerError({ code: 'backpressure.dropped', category: 'config', message: 'Dropped oldest by queue-bounded' }))
                } else {
                  throw createBrokerError({ code: 'backpressure.dropped', category: 'config', message: `Queue full (max ${max})`, details: { routeId, max, type: 'queue-bounded' } })
                }
              }
              return await executeNow(state, task)
            }
            case 'drop': {
              if (state.inFlight > 0) {
                throw createBrokerError({ code: 'backpressure.dropped', category: 'config', message: 'Drop policy: in-flight already exists', details: { routeId, type: 'drop' } })
              }
              return await executeNow(state, task)
            }
            case 'throttle': {
              const now = Date.now()
              const perSec = state.policy.perSec
              if (now - state.throttleWindowStart >= 1000) {
                state.throttleWindowStart = now
                state.throttleCount = 0
              }
              if (state.throttleCount >= perSec) {
                throw createBrokerError({ code: 'backpressure.dropped', category: 'config', message: `Throttle: max ${perSec}/sec`, details: { routeId, type: 'throttle' } })
              }
              state.throttleCount++
              return await executeNow(state, task)
            }
            case 'debounce': {
              const waitMs = state.policy.waitMs
              if (state.debounceTimer !== null) clearTimeout(state.debounceTimer)
              return await new Promise<T>((resolve, reject) => {
                state.debounceTimer = setTimeout(async () => {
                  state.debounceTimer = null
                  try { resolve(await executeNow(state, task)) } catch (err) { reject(err) }
                }, waitMs)
              })
            }
            case 'latest-only': {
              // Abort precedente — pending entries received abort
              for (const entry of state.pending) entry.controller.abort('backpressure.latest-only')
              state.pending = []
              return await executeNow(state, task)
            }
            case 'merge':
            case 'coalesce': {
              // V1 minimal: alias di latest-only (full coalescing rinviato a V1.x)
              for (const entry of state.pending) entry.controller.abort('backpressure.coalesce')
              state.pending = []
              return await executeNow(state, task)
            }
          }
          // Unreachable
          throw createBrokerError({ code: 'backpressure.unknown-policy', category: 'config', message: 'Unknown backpressure policy', details: { routeId } })
        },
        queueLength(routeId) {
          return states.get(routeId)?.pending.length ?? 0
        },
      }
    }
    ```
