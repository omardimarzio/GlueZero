    //
    // Composition root invocato da RouterBroker constructor:
    //   resolver, executor, httpGateway, outcomeCollector + strategies (config-derived).

    import type { BrokerEvent, EventTap, BrokerError } from '@sembridge/core'
    import type { GatewayConfig } from '@sembridge/gateway/http'
    import { HttpGateway } from '@sembridge/gateway/http'
    import {
      createRetryStrategy, createTimeoutStrategy, createIdempotencyStrategy,
      createDedupeStrategy, createBackpressureStrategy, createAuthStrategy, createCircuitBreakerStrategy,
    } from '@sembridge/gateway/http/strategies'
    import type { HttpGatewayStrategies } from '@sembridge/gateway/http'
    import { RouteResolver, type AmbiguousRouteEvent, type CompiledRoute } from './route-resolver'
    import { RouteExecutor } from './route-executor'
    import { OutcomeCollector, type PublishFn } from './outcome-collector'
    import { createHttpHandler } from './route-handlers/http-handler'
    import type { RoutingConfig } from './types/routing-config'

    export interface RouterEngineDeps {
      readonly mapper: { mapToShape: (canonical: unknown, outputMap: unknown) => unknown; mapToCanonical: (shape: unknown, schemaId: string) => unknown }
      readonly validator?: { validate: (schemaId: string, payload: unknown) => { ok: true } | { ok: false; issues: ReadonlyArray<{ message: string; path?: ReadonlyArray<string> }> } }
      readonly gatewayConfig?: GatewayConfig
      readonly routingConfig?: RoutingConfig
      readonly publishFn: PublishFn
      readonly tap?: EventTap
      readonly onAmbiguousRoutes?: (event: AmbiguousRouteEvent) => void
      readonly onCacheDeferred?: (event: { topic: string; routeId: string }) => void
    }

    export class RouterEngine {
      readonly resolver: RouteResolver
      readonly executor: RouteExecutor
      readonly httpGateway: HttpGateway
      readonly collector: OutcomeCollector
      readonly strategies: HttpGatewayStrategies

      constructor(deps: RouterEngineDeps) {
        const cfg = deps.gatewayConfig ?? {}

        // Strategy defaults config-derived
        this.strategies = {
          retry: createRetryStrategy({ maxAttempts: cfg.defaults?.retry?.maxAttempts ?? 3, baseDelayMs: cfg.defaults?.retry?.baseDelayMs ?? 300, maxDelayMs: cfg.defaults?.retry?.maxDelayMs ?? 10_000 }),
          timeout: createTimeoutStrategy(),
          idempotency: createIdempotencyStrategy({ headerName: cfg.defaults?.idempotency?.headerName ?? 'Idempotency-Key' }),
          dedupe: createDedupeStrategy(),
          backpressure: createBackpressureStrategy({ defaultPolicy: cfg.defaults?.backpressure ?? { type: 'queue-bounded', max: 100 } }),
          ...(cfg.auth && { auth: createAuthStrategy({ config: cfg.auth }) }),
          ...(cfg.circuitBreaker && cfg.circuitBreaker !== false && { circuitBreaker: createCircuitBreakerStrategy({ config: cfg.circuitBreaker }) }),
        }

        this.httpGateway = new HttpGateway(cfg)
        this.resolver = new RouteResolver({
          ...(deps.onAmbiguousRoutes && { onAmbiguousRoutes: deps.onAmbiguousRoutes }),
        })

        // http-handler factory richiede mapper + gateway + strategies + validator
        const httpHandler = createHttpHandler({
          gateway: this.httpGateway,
          mapper: deps.mapper,
          ...(deps.validator && { validator: deps.validator }),
          strategies: this.strategies,
        })

        this.executor = new RouteExecutor({
          httpHandler: (event, route, signal) => httpHandler(event, route, signal),
          resolveSubRoute: (id) => this.resolver.list().find((r) => r.id === id),
          ...(deps.tap && { tap: deps.tap }),
          ...(deps.onCacheDeferred && { onCacheDeferred: deps.onCacheDeferred }),
        })

        this.collector = new OutcomeCollector({
          publishFn: deps.publishFn,
          ...(deps.tap && { tap: deps.tap }),
        })
      }
    }

          mapper: { mapToShape: (c, m) => /* delega via inner — placeholder */ this.delegateMapToShape(c, m), mapToCanonical: (s, id) => this.delegateMapToCanonical(s, id) },
          ...(this.tap && { tap: this.tap }),
          publishFn: (topic, payload, options) => this.inner.publish(topic, payload, options as Parameters<typeof this.inner.publish>[2]),
          ...(config.gateway && { gatewayConfig: config.gateway }),
          ...(config.routing && { routingConfig: config.routing }),
          onAmbiguousRoutes: (event) => {
            this.inner.publish('routing.ambiguous', event, { source: { type: 'system', id: 'router' } })
          },
          onCacheDeferred: (event) => {
            // BLOCKER 2 fix (revision): TOPIC_REGEX in @sembridge/core/topic-matcher.ts NON consente hyphen.
            // Topic 'routing.composite.cache-deferred' fallirebbe validateTopic(). Rinominato a
            // 'routing.composite.deferred' (drop prefix 'cache-': il topic vive sotto routing.composite.*
            // quindi il prefisso 'cache-' è ridondante). L'identifier interno/callback name nei
            // moduli composite handler (plan 03-06) può restare 'cache-deferred' poiché è solo
            // identifier TS, non un topic pubblicato.
            this.inner.publish('routing.composite.deferred', event, { source: { type: 'system', id: 'router' } })
          },
        })

        // 3. Bootstrap routes da config
        if (config.routes) {
          for (const r of config.routes) this.engine.resolver.register(r)
        }
      }

      // === Pipeline §28 publish orchestration ===
      publish<T>(topic: string, payload: T, options: Parameters<MapperBroker['publish']>[2] = {}): void {
        const factory = startStep()
        const eventId = (options as { id?: string }).id ?? `route-${Date.now()}-${Math.random()}`

        // Step 8: route.resolved
        const matches = this.engine.resolver.resolve(topic, this.multipleRoutesPolicy)
        if (this.tap) {
          safeTapStep(this.tap, 'event.route.resolved' as PipelineStep, factory('event.route.resolved' as PipelineStep, eventId, topic, {
            metadata: { routeIds: matches.map((m) => m.id), policy: this.multipleRoutesPolicy },
          }))
        }

        // ROUTE-16 chiusura (D-67)
        if (matches.length === 0) {
          // Check requiresRoute via canonical schema
          const requiresRoute = this.checkRequiresRoute(topic)
          if (requiresRoute) {
            const failedTopic = this.deriveFailedTopic(topic)
            this.inner.publish(failedTopic, {
              error: {
                code: 'route.required.missing',
                category: 'config',
                message: `Topic "${topic}" requires a route but none registered (ROUTE-16)`,
                topic,
                eventId,
              },
              sourceEvent: { topic, eventId },
            }, { source: { type: 'system', id: 'router' } })
            return
          }
          // Default: delivery locale (D-67 default)
          this.inner.publish(topic, payload, options)
          return
        }

        // Step 9 + 10 per ogni route — async parallel
        for (const route of matches) {
          this.executeRoute(route, topic, payload, options, eventId)
        }
        // Local consumer ricevono comunque se almeno una route NON è 'local'
        // (D-65 default 'parallel': local + remote in parallelo)
        if (matches.some((r) => r.definition.type !== 'local')) {
          this.inner.publish(topic, payload, options)
        }
      }

      private async executeRoute<T>(route: CompiledRoute, topic: string, payload: T, options: unknown, eventId: string): Promise<void> {
        const event: BrokerEvent = {
          id: eventId,
          topic,
          payload,
          timestamp: Date.now(),
          source: (options as { source?: BrokerEvent['source'] }).source ?? { type: 'plugin', id: 'unknown' },
        } as BrokerEvent
        try {
          const outcome = await this.engine.executor.execute(route, event)
          this.engine.collector.collect(outcome, route, event)
        } catch (err) {
          // executor non dovrebbe throw — convertito a outcome.error.
          // Se throw inaspettato, log + publish failed di sicurezza.
          const safeError = isBrokerError(err) ? err : createBrokerError({
            code: 'route.executor.crashed',
            category: 'config',
            message: (err as Error).message ?? 'Unknown executor crash',
            routeId: route.id, topic, eventId,
            originalError: err as Error,
          })
          this.engine.collector.collect({ ok: false, error: safeError, routeId: route.id }, route, event)
        }
      }

      // === Route management (D-60) ===
      registerRoute(def: RouteDefinition, options: { ownerId?: string } = {}): RouteRegistration {
        return this.engine.resolver.register(def, options)
      }

      unregisterRoute(routeId: string): boolean {
        // Abort eventuali fetch in volo bound a routeId
        // (HttpGateway tracks by eventId, non routeId — best effort tracking)
        return this.engine.resolver.unregister(routeId)
      }

      // === Subscribe delegate (BLOCKER 3 revision fix) ===
      // RouterBroker DEVE esporre subscribe() come passthrough esplicito a MapperBroker.
      // Il test harness di 03-13 (router-harness.ts) e i 14 integration test dipendono

      // semantica F2 senza override.
      subscribe(
        ...args: Parameters<MapperBroker['subscribe']>
      ): ReturnType<MapperBroker['subscribe']> {
        return this.inner.subscribe(...args)
      }

      // === Plugin management (override per cascade D-86) ===
      registerPlugin(descriptor: PluginDescriptor): ReturnType<MapperBroker['registerPlugin']> {
        const result = this.inner.registerPlugin(descriptor)
        if (descriptor.routes) {
          for (const r of descriptor.routes) {
            try {
              this.engine.resolver.register(r, { ownerId: descriptor.id })
            } catch (err) {
              // Log isolated — se route fail, plugin register continua
              this.tap?.onPipelineStep?.('event.route.registration.failed' as PipelineStep, { eventId: 'register', topic: 'system', step: 'register' as PipelineStep, timestamp: Date.now(), durationMs: 0 } as never)
            }
          }
        }
        return result
      }

      async unregisterPlugin(id: string): Promise<void> {
        // LIFE-02 ext F3 cascade (D-86) — try/catch isolato per ogni step (pattern F2)
        const recordCascadeError = (step: string, err: unknown): void => {
          // Log silently — pattern F2 (vedi mapper unregisterPlugin)

        try { this.engine.httpGateway.abortInFlightByOwner(id) } catch (err) { recordCascadeError('gateway.abort', err) }
      }

      // === Public API delegate (canonical schema, transform, alias — F2) ===
      registerCanonicalSchema(...args: Parameters<MapperBroker['registerCanonicalSchema']>): ReturnType<MapperBroker['registerCanonicalSchema']> {
        return this.inner.registerCanonicalSchema(...args)
      }

      // === Helpers ===
      private deriveFailedTopic(topic: string): string {
        const idx = topic.lastIndexOf('.')
        if (idx === -1) return `${topic}.failed`
        const suffix = topic.slice(idx + 1)
        if (suffix === 'requested') return `${topic.slice(0, idx)}.failed`
        return `${topic}.failed`
      }

      // === ROUTE-16 helpers (BLOCKER 4 revision fix) ===
      //
      // BLOCKER 4 fix: la versione precedente di checkRequiresRoute faceva un silent
      // fallback a `false` se l'accesso a `inner.canonicalRegistry.getForTopic` falliva
      // — questo nascondeva l'opt-in D-67 (ROUTE-16) trasformandolo in NO-OP silenzioso.
      //
      // Realtà API F2 verificata in revisione (packages/mapper/src/broker-mapper-wrapper.ts
      // lines 206-211 + canonical-registry.ts):
      //   - `MapperBroker.canonicalRegistry` è `private readonly` (non esposto)
      //   - `CanonicalRegistry` espone `has(id)` / `get(id)` / `list()` — NON ha `getForTopic(topic)`
      //   - Il legame topic → CanonicalSchemaId NON è una mapping table pubblica in F2
      //
      // Strategia (Opzione A revision_context — getter sul RouterBroker che ISOLA il cast
      // UNA volta + surface esplicito):
      //   1. Constructor: bind del registry via cast tipato esplicito; se manca → throw
      //      `BrokerError 'router.canonical-registry.unavailable'` al boot. NIENTE silent
      //      fallback. Il consumer vede il fallimento al `createRouterBroker(config)`,
      //      non quando un evento ROUTE-16 prova a partire e silenziosamente esegue
      //      delivery locale.
      //   2. Per derivare il schemaId dal topic, usiamo la convenzione PRD §11
      //      "<entity>.<action>.<status>": il primo segmento del topic = entity, che è
      //      anche per convenzione il `CanonicalSchemaId` (es. 'weather.requested' →
      //      schema 'weather'). Se il consumer usa schemaId differenti dal primo
      //      segmento, può ovverridare via `RoutingConfig.requiresRouteTopics: string[]`
      //      come opt-in esplicito (lista di topic che `requiresRoute: true`); questo
      //      bypassa completamente il lookup canonical e chiude il blocker senza
      //      dipendere dall'accesso al registry private.
      //   3. Test loud: oltre al test positive (publish .failed con
      //      'route.required.missing'), il test di Plan 03-12 include un test che
      //      verifica il throw esplicito al boot se `inner.canonicalRegistry` non è
      //      accessibile (mock di MapperBroker senza canonicalRegistry).
      //
      // Documentato come D-100 in CONTEXT.md (revision iter 1): "RouterBroker isola
      // l'accesso al CanonicalRegistry private di F2 in un getter dedicato; F2 può
      // esporre un helper public in F6 senza breaking change al RouterBroker".
      private getCanonicalSchemaForTopic(topic: string): { requiresRoute?: boolean } | undefined {
        // Cast tipato esplicito al canonicalRegistry private — bound al construct,
        // verificato non-undefined in constructor (vedi `bindCanonicalRegistry` sotto).
        const registry = this.boundCanonicalRegistry
        if (!registry) {
          // Difensivo — il constructor avrebbe già throw, ma manteniamo il guard.
          return undefined
        }
        // Convenzione PRD §11 <entity>.<action>.<status>: primo segmento = schemaId.
        // CanonicalSchemaId è branded — riusiamo la string come unbranded e
        // CanonicalRegistry.has/get accetta CanonicalSchemaId via cast (pattern F2
        // public-factory.ts riusa lo stesso approccio per parsedSchemas).
        const segments = topic.split('.')
        const entityCandidate = segments[0]
        if (!entityCandidate) return undefined
        // Override esplicito via RoutingConfig.requiresRouteTopics (opt-in B):
        // se il consumer ha dichiarato il topic in lista, ritorniamo uno schema
        // sintetico { requiresRoute: true } SENZA passare dal registry.
        if (this.explicitRequiresRouteTopics.has(topic)) {
          return { requiresRoute: true }
        }
        // Lookup nominale via primo segmento (convenzione PRD §11). `as never` per
        // bypass del CanonicalSchemaId brand: se lo schema esiste, ritorna il flag
        // augmentato D-95; se no, ritorna undefined → caller usa default local delivery.
        try {
          return registry.get(entityCandidate as never) as { requiresRoute?: boolean } | undefined
        } catch (err) {
          // Loud failure: registry.get NON dovrebbe throw per id inesistente
          // (CanonicalRegistry.get usa Map.get che ritorna undefined). Se throw =>
          // problema di integrazione F2 → propaga.
          throw createBrokerError({
            code: 'router.canonical-registry.lookup.failed',
            category: 'config',
            message: `RouterBroker: canonicalRegistry.get failed for topic "${topic}"`,
            details: { topic, entityCandidate },
            ...(err instanceof Error && { originalError: err }),
          })
        }
      }

      private checkRequiresRoute(topic: string): boolean {
        const schema = this.getCanonicalSchemaForTopic(topic)
        return schema?.requiresRoute === true
      }

      private delegateMapToShape(canonical: unknown, outputMap: unknown): unknown {
        const inner = this.inner as unknown as { mapper?: { mapToShape?: (c: unknown, m: unknown) => unknown } }
        return inner.mapper?.mapToShape?.(canonical, outputMap) ?? canonical
      }

      private delegateMapToCanonical(shape: unknown, schemaId: string): unknown {
        const inner = this.inner as unknown as { mapper?: { mapToCanonical?: (s: unknown, id: string) => unknown } }
        return inner.mapper?.mapToCanonical?.(shape, schemaId) ?? shape
      }
    }
