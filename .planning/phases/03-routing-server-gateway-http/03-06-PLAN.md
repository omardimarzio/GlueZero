    ```typescript
    import { createBrokerError } from '@gluezero/core'
    import type { BrokerEvent } from '@gluezero/core'
    import type { CompiledRoute } from '../route-resolver'
    import type { RouteCompositeDefinition } from '../types/route-definition'
    import type { RouteOutcome } from '../types/route-outcome'

    /**
     * Handler per route `'composite'` — workflow F3 (ROUTE-05).
     *
     * F3 behavior (Open Question Q3 RESEARCH): skip cache check (adapter F6) + warning UNA volta in dev mode.
     * Composite F3 = alias per http route — il vero workflow check-cache → http → update-cache arriva con F6.
     *
     * Lo step 'http' del composite delega all'httpHandler (passato come dependency injection).
     * Lo step 'publish' è gestito dall'OutcomeCollector a valle (plan 03-07).
     */
    export interface CompositeHandlerDeps {
      readonly httpHandler: (event: BrokerEvent, subRoute: CompiledRoute) => Promise<RouteOutcome>
      readonly resolveSubRoute: (id: string) => CompiledRoute | undefined
      /** Callback opt-in dev-mode warning. Default no-op. */
      readonly onCacheDeferred?: (event: { topic: string; routeId: string }) => void
    }

    export function createCompositeHandler(deps: CompositeHandlerDeps) {
      let cacheWarnEmitted = false
      return async function compositeHandler(
        event: BrokerEvent,
        route: CompiledRoute,
      ): Promise<RouteOutcome> {
        const def = route.definition as RouteCompositeDefinition
        // Find first 'http' step
        const httpStep = def.steps.find((s) => s.type === 'http')
        if (!httpStep || !httpStep.route) {
          return {
            ok: false,
            error: createBrokerError({
              code: 'route.composite.no-http',
              category: 'config',
              message: 'Composite route requires at least one http step in F3 (cache adapter F6 deferred)',
              routeId: route.id,
              topic: event.topic,
              eventId: event.id,
            }),
            routeId: route.id,
          }
        }
        // Warn cache-deferred una volta
        if (def.steps.some((s) => s.type === 'cache') && !cacheWarnEmitted) {
          cacheWarnEmitted = true
          deps.onCacheDeferred?.({ topic: event.topic, routeId: route.id })
        }
        // Resolve sub-route + delegate
        const subRoute = deps.resolveSubRoute(httpStep.route)
        if (!subRoute) {
          return {
            ok: false,
            error: createBrokerError({
              code: 'route.composite.subroute-missing',
              category: 'config',
              message: `Composite step refers to unknown route "${httpStep.route}"`,
              routeId: route.id,
              topic: event.topic,
              eventId: event.id,
            }),
            routeId: route.id,
          }
        }
        return await deps.httpHandler(event, subRoute)
      }
    }
    ```

    ```typescript
    export class RouteExecutor {
      private readonly inFlight = new Map<string, InFlightEntry>()
      private readonly composite: ReturnType<typeof createCompositeHandler>
      private readonly tap: EventTap | undefined

      constructor(private readonly deps: RouteExecutorDeps) {
        this.composite = createCompositeHandler({
          httpHandler: (e, r) => this.deps.httpHandler(e, r, this.getOrCreateController(e.id, r).signal),
          resolveSubRoute: deps.resolveSubRoute,
          ...(deps.onCacheDeferred && { onCacheDeferred: deps.onCacheDeferred }),
        })
        this.tap = deps.tap
      }

      async execute(route: CompiledRoute, event: BrokerEvent): Promise<RouteOutcome> {
        const factory = startStep()
        const controller = this.getOrCreateController(event.id, route)
        let outcome: RouteOutcome
        try {
          switch (route.definition.type) {
            case 'local':
              outcome = localHandler(event, route)
              break
            case 'http':
              outcome = await this.deps.httpHandler(event, route, controller.signal)
              break
            case 'cache':
              outcome = cacheHandler(event, route)
              break
            case 'composite':
              outcome = await this.composite(event, route)
              break
            default: {
              const t = (route.definition as { type: string }).type
              outcome = {
                ok: false,
                routeId: route.id,
                error: createBrokerError({
                  code: 'route.type.unknown',
                  category: 'config',
                  message: `Unknown route type: "${t}"`,
                  routeId: route.id,
                  topic: event.topic,
                  eventId: event.id,
                }),
              }
            }
          }
        } finally {
          this.inFlight.delete(event.id)
        }
        // Emette tap step 9
        if (this.tap !== undefined) {
          safeTapStep(
            this.tap,
            'event.route.executed' as PipelineStep,
            factory('event.route.executed' as PipelineStep, event.id, event.topic, {
              metadata: { routeId: route.id, routeType: route.definition.type, ok: outcome.ok },
            }),
          )
        }
        return outcome
      }

      abortInFlight(eventId: string, reason: string = 'gateway.aborted'): boolean {
        const entry = this.inFlight.get(eventId)
        if (!entry) return false
        entry.controller.abort(reason)
        return true
      }

      abortInFlightByOwner(ownerId: string, reason: string = 'plugin.unregistered'): number {
        let count = 0
        for (const [, entry] of this.inFlight) {
          if (entry.ownerId === ownerId) {
            entry.controller.abort(reason)
            count++
          }
        }
        return count
      }

      inFlightCount(): number {
        return this.inFlight.size
      }

      private getOrCreateController(eventId: string, route: CompiledRoute): AbortController {
        const existing = this.inFlight.get(eventId)
        if (existing) return existing.controller
        const controller = new AbortController()
        this.inFlight.set(eventId, {
          controller,
          ownerId: route.ownerId,
          routeId: route.id,
        })
        return controller
      }
    }
    ```
