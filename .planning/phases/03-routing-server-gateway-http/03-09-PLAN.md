    const DEFAULT_RETRY_STATUSES = new Set([408, 429])
    const SERVER_ERROR_MIN = 500
    const SERVER_ERROR_MAX = 599

    function isRetriableStatus(status: number, customRetryOn: ReadonlySet<number> | undefined): boolean {
      if (customRetryOn !== undefined) return customRetryOn.has(status)
      if (DEFAULT_RETRY_STATUSES.has(status)) return true
      if (status >= SERVER_ERROR_MIN && status <= SERVER_ERROR_MAX) return true
      return false
    }

    export function createRetryStrategy(options: RetryStrategyOptions = {}): RetryStrategy {
      const maxAttempts = options.maxAttempts ?? 3
      const baseDelayMs = options.baseDelayMs ?? 300
      const maxDelayMs = options.maxDelayMs ?? 10_000
      const customRetryOn = options.retryOnStatuses ? new Set(options.retryOnStatuses) : undefined

      return {
        shouldRetry(response, error, attempt) {
          if (attempt >= maxAttempts) return false
          // Network error → retry
          if (error !== undefined && response === undefined) return true
          // Response status check
          if (response !== undefined) {
            return isRetriableStatus(response.status, customRetryOn)
          }
          return false
        },
        delayMs(attempt, retryAfterHeader) {
          // D-69: rispetta Retry-After se presente (priority over jitter)
          if (retryAfterHeader) {
            const parsed = parseRetryAfter(retryAfterHeader)
            if (parsed !== undefined) return Math.min(parsed, MAX_BACKOFF_MS)
          }
          // Full jitter formula esatta (Pitfall 2 fix)
          const exponential = Math.min(maxDelayMs, baseDelayMs * Math.pow(2, attempt))
          const jittered = exponential * (0.5 + Math.random() * 0.5)
          return Math.floor(jittered)
        },
      }
    }
    ```

    // idempotency-strategy.ts — AutoIdempotency (D-70, SEC-03 + Pitfall 3 fix).
    //
    // Genera Idempotency-Key (nanoid) PERSISTENTE PER eventId — riusato sui retry.
    // Il server è responsabile di deduplicare per la chiave (precondizione DOC-04).

    import { nanoid } from 'nanoid'
    import type { IdempotencyStrategy } from '../types/http-strategies'

    export interface IdempotencyStrategyOptions {
      readonly headerName?: string
      readonly tokenFactory?: () => string
      /** LRU cap per evitare memory leak in long-running clients (default 1000). */
      readonly maxEventsTracked?: number
    }

    export function createIdempotencyStrategy(options: IdempotencyStrategyOptions = {}): IdempotencyStrategy {
      const headerName = options.headerName ?? 'Idempotency-Key'
      const tokenFactory = options.tokenFactory ?? (() => nanoid())
      const maxTracked = options.maxEventsTracked ?? 1000
      const tokenByEventId = new Map<string, string>()

      return {
        generate(eventId) {
          const existing = tokenByEventId.get(eventId)
          if (existing !== undefined) return existing
          const token = tokenFactory()
          tokenByEventId.set(eventId, token)
          // LRU bounded — drop oldest se supera maxTracked
          if (tokenByEventId.size > maxTracked) {
            const firstKey = tokenByEventId.keys().next().value
            if (firstKey !== undefined) tokenByEventId.delete(firstKey)
          }
          return token
        },
        headerName() {
          return headerName
        },
      }
    }
    ```
