    ```typescript
    export function combineSignals(...signals: ReadonlyArray<AbortSignal | undefined>): AbortSignal {
      const real = signals.filter((s): s is AbortSignal => s !== undefined)
      const Native = (AbortSignal as unknown as { any?: (s: AbortSignal[]) => AbortSignal }).any
      if (typeof Native === 'function') return Native.call(AbortSignal, real)
      const composite = new AbortController()
      for (const sig of real) {
        if (sig.aborted) { composite.abort(sig.reason); break }
        sig.addEventListener('abort', () => composite.abort(sig.reason), { once: true })
      }
      return composite.signal
    }
    ```

    ```typescript
    export const MAX_BACKOFF_MS = 60_000

    export function parseRetryAfter(headerValue: string | null, now: number = Date.now()): number | undefined {
      if (!headerValue) return undefined
      const trimmed = headerValue.trim()
      const seconds = Number(trimmed)
      if (!Number.isNaN(seconds) && seconds >= 0) return Math.min(seconds * 1000, MAX_BACKOFF_MS)
      const dateMs = Date.parse(trimmed)
      if (!Number.isNaN(dateMs)) {
        const delta = dateMs - now
        if (delta < 0) return 0
        return Math.min(delta, MAX_BACKOFF_MS)
      }
      return undefined
    }
    ```

    ```typescript
    import type { GatewayContext, GatewayMiddleware } from './types/http-strategies'

    /**
     * Koa-style compose async middleware (Pattern 3 RESEARCH).
     * Minimal allocation per request — closures bound al construct, ctx mutabile in-place.
     */
    export function compose(middlewares: readonly GatewayMiddleware[]): (ctx: GatewayContext) => Promise<void> {
      return async (ctx) => {
        let index = -1
        const dispatch = async (i: number): Promise<void> => {
          if (i <= index) throw new Error('next() called multiple times')
          index = i
          const fn = middlewares[i]
          if (!fn) return
          await fn(ctx, () => dispatch(i + 1))
        }
        return dispatch(0)
      }
    }
    ```

    ```typescript
    export class HttpGateway {
      private readonly inFlight = new Map<string, InFlightEntry>()

      constructor(private readonly config: GatewayConfig = {}) {}

      async execute(
        request: HttpRequestSpec,
        route: { id: string; ownerId?: string },
        event: BrokerEvent,
        externalSignal: AbortSignal | undefined,
        strategies: HttpGatewayStrategies,
      ): Promise<HttpResponseSpec> {
        // SEC-05: pre-fetch allowlist validation
        validateAgainstAllowlist(request.url, this.config.allowlist, {
          routeId: route.id, topic: event.topic, eventId: event.id,
        })

        // Auth header injection (D-72) — single-flight refresh handled by AuthStrategy
        let headers: Record<string, string> = { ...request.headers }
        if (strategies.auth) {
          const token = await strategies.auth.getToken()
          if (token !== undefined) headers['Authorization'] = `Bearer ${token}`
        }

        // Idempotency-Key (D-70) — UNA volta, persistente su retry (Pitfall 3)
        if (strategies.idempotency && (request.method === 'POST' || request.method === 'PATCH' || request.method === 'PUT' || request.method === 'DELETE')) {
          headers[strategies.idempotency.headerName()] = strategies.idempotency.generate(event.id)
        }

        // Combined signal (D-77 + Pitfall 4)
        const ownController = new AbortController()
        this.inFlight.set(event.id, { controller: ownController, ownerId: route.ownerId, routeId: route.id })

        const timeoutMs = (typeof this.config.defaults?.timeout === 'number') ? this.config.defaults.timeout : 30_000
        const timeoutSignal = strategies.timeout?.signal(timeoutMs) ?? AbortSignal.timeout(timeoutMs)
        const fetchSignal = combineSignals(externalSignal, ownController.signal, timeoutSignal)

        const finalRequest = { ...request, headers }

        try {
          // Circuit breaker check (D-99)
          if (strategies.circuitBreaker && !strategies.circuitBreaker.canExecute(route.id)) {
            throw createBrokerError({
              code: 'circuit.open', category: 'config',
              message: `Circuit breaker open for route "${route.id}"`,
              routeId: route.id, topic: event.topic, eventId: event.id,
            })
          }

          // Retry loop — delegates to RetryStrategy.shouldRetry/delayMs
          let attempt = 0
          let lastError: Error | undefined
          let lastResponse: Response | undefined
          const maxAttempts = strategies.retry ? (this.config.defaults?.retry?.maxAttempts ?? 3) : 1

          while (attempt < maxAttempts) {
            attempt++
            try {
              const response = await this.fetchOnce(finalRequest, fetchSignal, strategies)
              lastResponse = response
              if (response.ok || !strategies.retry?.shouldRetry(response, undefined, attempt)) {
                strategies.circuitBreaker?.recordSuccess(route.id)
                return await this.parseResponse(response, route.id, event)
              }
              // attesa retry
              if (attempt < maxAttempts) {
                const retryAfter = response.headers.get('Retry-After')
                const delay = strategies.retry.delayMs(attempt, retryAfter)
                await new Promise<void>((res) => setTimeout(res, Math.min(delay, MAX_BACKOFF_MS)))
              }
            } catch (err) {
              lastError = err as Error
              if (!strategies.retry?.shouldRetry(undefined, lastError, attempt) || attempt >= maxAttempts) break
              await new Promise<void>((res) => setTimeout(res, strategies.retry!.delayMs(attempt)))
            }
          }

          strategies.circuitBreaker?.recordFailure(route.id)

          // outcome finale dopo retry exhausted
          if (lastResponse) {
            return await this.responseToSpec(lastResponse, route.id, event, attempt)
          }
          // Network error
          throw createBrokerError({
            code: this.classifyError(lastError, ownController.signal),
            category: 'network',
            message: lastError?.message ?? 'Network error',
            routeId: route.id, topic: event.topic, eventId: event.id,
            ...(lastError && { originalError: lastError }),
            details: { attemptCount: attempt },
          })
        } finally {
          this.inFlight.delete(event.id)
        }
      }

      abortInFlight(eventId: string, reason = 'gateway.aborted'): boolean {
        const entry = this.inFlight.get(eventId)
        if (!entry) return false
        entry.controller.abort(reason)
        return true
      }

      abortInFlightByOwner(ownerId: string, reason = 'plugin.unregistered'): number {
        let count = 0
        for (const [, entry] of this.inFlight) {
          if (entry.ownerId === ownerId) { entry.controller.abort(reason); count++ }
        }
        return count
      }

      inFlightCount(): number {
        return this.inFlight.size
      }

      private async fetchOnce(req: HttpRequestSpec, signal: AbortSignal, strategies: HttpGatewayStrategies): Promise<Response> {
        const init: RequestInit = {
          method: req.method, headers: req.headers, signal, redirect: 'manual',
          ...(req.body !== undefined && { body: req.body }),
        }
        const response = await fetch(req.url, init)
        // Pitfall 7: post-redirect re-validation
        if (response.status >= 300 && response.status < 400) {
          const location = response.headers.get('Location')
          if (location) {
            validateAgainstAllowlist(new URL(location, req.url).href, this.config.allowlist)
            // se passa: refetch manualmente con stessi headers (preserve Idempotency-Key)
            return await fetch(location, init)
          }
        }
        return response
      }

      private async parseResponse(response: Response, routeId: string, event: BrokerEvent): Promise<HttpResponseSpec> {
        let body: unknown
        try { body = await response.json() } catch { body = null }
        return {
          ok: response.ok,
          status: response.status,
          headers: Object.fromEntries(response.headers.entries()),
          body,
        }
      }

      private async responseToSpec(response: Response, routeId: string, event: BrokerEvent, attempt: number): Promise<HttpResponseSpec> {
        return await this.parseResponse(response, routeId, event)
      }

      private classifyError(err: Error | undefined, signal: AbortSignal): 'gateway.timeout' | 'gateway.network' | 'gateway.aborted' {
        if (signal.aborted) {
          const reason = String(signal.reason ?? '')
          if (reason.includes('timeout') || reason === 'gateway.timeout') return 'gateway.timeout'
          return 'gateway.aborted'
        }
        return 'gateway.network'
      }
    }
    ```

    Factory:
    ```typescript
    export function createHttpHandler(deps: HttpHandlerDeps) {
      return async function httpHandler(
        event: BrokerEvent,
        route: CompiledRoute,
        signal: AbortSignal,
      ): Promise<RouteOutcome> {
        const def = route.definition as RouteHttpDefinition
        try {
          // Step 1: build request via mapper.mapToShape
          const queryShape = def.request.queryMap !== undefined
            ? deps.mapper.mapToShape(event.payload, def.request.queryMap) as Record<string, unknown>
            : {}
          const bodyShape = def.request.bodyMap !== undefined
            ? deps.mapper.mapToShape(event.payload, def.request.bodyMap)
            : undefined

          const queryString = new URLSearchParams(
            Object.fromEntries(
              Object.entries(queryShape ?? {}).map(([k, v]) => [k, String(v)]),
            ),
          ).toString()
          const fullUrl = queryString ? `${def.request.url}?${queryString}` : def.request.url

          const httpRequest = {
            method: def.request.method,
            url: fullUrl,
            headers: { 'Content-Type': 'application/json', Accept: 'application/json' },
            ...(bodyShape !== undefined && { body: JSON.stringify(bodyShape) }),
          }

          // Step 2: delegate al gateway
          const response = await deps.gateway.execute(
            httpRequest,
            { id: route.id, ...(route.ownerId !== undefined && { ownerId: route.ownerId }) },
            event, signal, deps.strategies,
          )

          if (!response.ok) {
            const code = response.status >= 500 ? 'gateway.5xx' : 'gateway.4xx'
            return {
              ok: false,
              routeId: route.id,
              error: createBrokerError({
                code, category: 'network',
                message: `HTTP ${response.status}`,
                routeId: route.id, topic: event.topic, eventId: event.id,
                details: { httpStatus: response.status, body: response.body },
              }),
            }
          }

          // Step 3: mapper response → canonical (D-97)
          const canonicalResponse = deps.mapper.mapToCanonical(response.body, def.response.canonical)

          // Step 4: response validation opt-in (VAL-05)
          if (deps.validator) {
            const validationResult = deps.validator.validate(def.response.canonical, canonicalResponse)
            if (!validationResult.ok) {
              return {
                ok: false,
                routeId: route.id,
                error: createBrokerError({
                  code: 'response.validation.failed',
                  category: 'validation',
                  message: 'Server response failed canonical schema validation',
                  routeId: route.id, topic: event.topic, eventId: event.id,
                  details: { schemaId: def.response.canonical, issues: validationResult.issues },
                }),
              }
            }
          }

          return {
            ok: true,
            canonicalPayload: canonicalResponse,
            routeId: route.id,
            metadata: { httpStatus: response.status, attemptCount: 1, origin: 'remote' as const },
          }
        } catch (err) {
          // Errori dal gateway (gateway.url.forbidden, gateway.network, ...) sono BrokerError
          if (isBrokerError(err)) {
            return { ok: false, error: err, routeId: route.id }
          }
          return {
            ok: false,
            routeId: route.id,
            error: createBrokerError({
              code: 'gateway.network', category: 'network',
              message: (err as Error).message ?? 'Unknown gateway error',
              routeId: route.id, topic: event.topic, eventId: event.id,
              originalError: err as Error,
            }),
          }
        }
      }
    }
    ```
