**Step 1 — Crea `packages/gateway/src/sse-ws/realtime-broker.ts`** basato su PATTERNS.md §2.11:

```typescript
import type { PluginDescriptor, Subscription } from '@sembridge/core'
import { RouterBroker, type RouterBrokerConfig } from '@sembridge/routing'
import type { BackpressureStrategy } from '../http/strategies/backpressure-strategy'
import { RealtimeChannelManager, type RealtimeChannelManagerDeps } from './realtime-channel-manager'
import type { RealtimeConfig } from './types/realtime-config'
import type { RealtimeChannelDef } from './types/realtime-channel-def'

/**
 * Configurazione RealtimeBroker — accetta tutto il `RouterBrokerConfig` di F3 + sezione F4.
 *
 * Pattern declaration merging: `realtime?: RealtimeConfig` aggiunto via `sse-ws/augment.ts`
 * (plan 04-01 D-103). Per chiarezza export-side ridichiariamo il super-set come interface.
 */
export interface RealtimeBrokerConfig extends RouterBrokerConfig {
  readonly realtime?: RealtimeConfig
}

/**
 * RealtimeBroker — composition wrapper di RouterBroker per F4 SSE/WS (D-101).
 *
 * Pattern identico a `RouterBroker` (router-broker-wrapper.ts:105-226):
 *   - inner: `RouterBroker` (F3) — delegato per pub/sub/lifecycle base + routing + http gateway
 *   - manager: `RealtimeChannelManager` — N canali SSE/WS indicizzati per name (plan 04-07)
 *   - publish/subscribe → delegate a inner (composition, NO modify)
 *   - registerPlugin/unregisterPlugin → override per cascade auto-register channels (D-103) +
 *     cascade cleanup (D-112)
 *   - connectRealtime/disconnectRealtime → API surface F4 pubblica (PRD §16.2)
 *
 * **Vincolo D-83 strict (carryover F3 → F4)**: ZERO modifiche a `packages/core/`,
 * `packages/mapper/`, `packages/routing/`, `packages/gateway/src/http/` runtime. F4 vive
 * SOLO in `packages/gateway/src/sse-ws/` + `packages/gateway/src/sse-ws/augment.ts`
 * (declaration merging dei tipi).
 *
 * @example
 * ```ts
 * const broker = createRealtimeBroker({
 *   realtime: {
 *     channels: [
 *       { name: 'orders', mode: 'auto', buildUrl: async () => `/events?token=${await getToken()}` }
 *     ]
 *   }
 * })
 * await broker.connectRealtime({ name: 'notifications', mode: 'sse', url: '/notifications' })
 * broker.subscribe('orders.created', (ev) => { ... })
 * ```
 */
export class RealtimeBroker {
  private readonly inner: RouterBroker
  private readonly manager: RealtimeChannelManager

  constructor(config: RealtimeBrokerConfig = {}) {
    // 1. Compose RouterBroker (F3) — pattern identico RouterBroker → MapperBroker (D-83 chain).
    this.inner = new RouterBroker(config)

    // 2. Build Manager con publishFn legato all'inner (pipeline §28 step 1 ingress D-113).
    //    Il publishFn riceve un BrokerEvent completo costruito dall'adapter; lo trasformiamo
    //    in chiamata a inner.publish(topic, payload, options) preservando source.
    const managerDeps: RealtimeChannelManagerDeps = {
      publishFn: (event) => {
        // W-1 fix — preservazione source.type='server' end-to-end (chiusura D-113 ingress).
        //
        // Verifica preventiva durante l'execution (read_first):
        //   1. Leggere `packages/routing/src/router-broker-wrapper.ts` cerca firma `publish(`.
        //   2. Se accetta options.source/id come terzo arg → usa il pattern qui sotto.
        //   3. Se NON li accetta → identificare il meccanismo F1/F2/F3 per pubblicare BrokerEvent
        //      "as-is" (probabilmente c'è un overload `publish(event: BrokerEvent)` o un
        //      `publishRaw` interno; cercare `publish.*BrokerEvent` o `publishEvent` in
        //      core/bus.ts + router-broker-wrapper.ts).
        //   4. Se nessuna delle due opzioni esiste, è BLOCKER: aggiornare il piano con un fix
        //      su routing (richiede deroga D-83 — escalare).
        //
        // Pattern attuale (da verificare vivo):
        this.inner.publish(event.topic, event.payload as never, {
          source: event.source,
          id: event.id,
        } as Parameters<RouterBroker['publish']>[2])
      },
    }

    // Backpressure shared per-broker (riuso F3 pattern — opt-in via config.realtime.defaults)
    if (this.tryGetSharedBackpressure(config) !== undefined) {
      ;(managerDeps as { backpressure?: BackpressureStrategy }).backpressure = this.tryGetSharedBackpressure(config)
    }

    this.manager = new RealtimeChannelManager(managerDeps)

    // 3. Bootstrap channels da config (D-102, analogo `routes` di F3 plan 03-12).
    if (config.realtime?.channels) {
      for (const def of config.realtime.channels) {
        // Fire-and-forget connect — pattern Promise.catch difensivo come router-broker-wrapper.ts:316.
        this.manager.connect(def, 'system').catch(() => {
          // No-op: errore già pubblicato come system.realtime.disconnected dal manager/adapter.
        })
      }
    }
  }

  // ============================================================
  // Realtime API (D-102, PRD §16.2)
  // ============================================================

  /**
   * Connetti un canale realtime (D-102). Se `def.name` già registrato → throw `realtime.channel.duplicate`.
   *
   * @example
   * ```ts
   * await broker.connectRealtime({ name: 'orders', mode: 'auto', url: '/events' })
   * ```
   */
  async connectRealtime(def: RealtimeChannelDef): Promise<void> {
    return this.manager.connect(def, 'system')
  }

  /**
   * Disconnetti un canale (`name` omesso = tutti — D-102).
   *
   * @example
   * ```ts
   * broker.disconnectRealtime('orders')  // singolo
   * broker.disconnectRealtime()           // tutti + visibility teardown
   * ```
   */
  disconnectRealtime(name?: string): void {
    return this.manager.disconnect(name)
  }

  // ============================================================
  // Plugin management (override per cascade D-103 + D-112)
  // ============================================================

  /**
   * Registra un plugin — delegate a `RouterBroker.registerPlugin` + auto-register
   * `descriptor.realtimeChannels` con `ownerId = descriptor.id` (D-103).
   *
   * Pattern try/catch isolato identico a `RouterBroker.registerPlugin`
   * (router-broker-wrapper.ts:437-449): un canale fallito NON impedisce il register
   * del plugin, solo logga (manager publica system.realtime.disconnected internamente).
   */
  async registerPlugin(descriptor: PluginDescriptor): Promise<void> {
    await this.inner.registerPlugin(descriptor)
    if (descriptor.realtimeChannels && descriptor.realtimeChannels.length > 0) {
      for (const def of descriptor.realtimeChannels) {
        try {
          await this.manager.connect(def, descriptor.id)
        } catch (err) {
          // W-5 fix — niente silent catch: emit `system.warn` con dettagli per audit/debug.
          // Il plugin viene comunque registrato (graceful degrade pattern F3); il consumer
          // può sottoscriversi a `system.warn` per loggare/alertare.
          this.inner.publish('system.warn', {
            plugin: descriptor.id,
            channel: def.name,
            reason: 'realtime-channel-register-failed',
            error: err instanceof Error ? err.message : String(err),
          } as never, {
            source: { type: 'system', id: 'realtime-broker', name: 'register-plugin' },
          } as Parameters<RouterBroker['publish']>[2])
        }
      }
    }
  }

  /**
   * Unregister plugin — cascade D-112 (estende D-86 di F3).
   *
   * Sequenza con try/catch isolato per ogni step (pattern router-broker-wrapper.ts:463-485):
   *   1. inner.unregisterPlugin (F3 cascade routes + http abort + F2 cascade + F1 unsub)
   *   2. manager.disconnectByOwner (chiude canali realtime registrati dal plugin)
   */
  async unregisterPlugin(id: string): Promise<void> {
    try {
      await this.inner.unregisterPlugin(id)
    } catch {
      /* pattern F3 silent — un fail in F3 cascade NON deve bloccare F4 cleanup */
    }
    try {
      this.manager.disconnectByOwner(id)
    } catch {
      /* pattern F3 silent */
    }
  }

  // ============================================================
  // Public API delegate (publish/subscribe/route/canonical — passthrough)
  // ============================================================

  /** Delegate `inner.publish` — pattern router-broker-wrapper.ts:422-424. */
  publish(...args: Parameters<RouterBroker['publish']>): ReturnType<RouterBroker['publish']> {
    return this.inner.publish(...args)
  }

  /** Delegate `inner.subscribe`. */
  subscribe(...args: Parameters<RouterBroker['subscribe']>): Subscription {
    return this.inner.subscribe(...args)
  }

  /** Delegate `inner.registerRoute`. */
  registerRoute(...args: Parameters<RouterBroker['registerRoute']>): ReturnType<RouterBroker['registerRoute']> {
    return this.inner.registerRoute(...args)
  }

  /** Delegate `inner.unregisterRoute`. */
  unregisterRoute(...args: Parameters<RouterBroker['unregisterRoute']>): ReturnType<RouterBroker['unregisterRoute']> {

 * Harness per integration test F4 (D-119 6 scenari obbligatori).

 * Pattern analog `router-harness.ts` di F3 — collect events, mock SSE/WS,
 * reset deterministico per test isolation.
 *

import { createRealtimeBroker, type RealtimeBroker } from '../public-factory'
import type { RealtimeBrokerConfig } from '../realtime-broker'
import type { RealtimeChannelDef } from '../types/realtime-channel-def'

  /** Topic da pre-collezionare. Default: tutti gli eventi system.realtime.* + network.error. */
  readonly collectTopics?: readonly string[]
}

export interface CollectedEvent {
  readonly topic: string
  readonly payload: unknown
  readonly source?: { readonly type: string; readonly name?: string }
  readonly id?: string
  readonly timestamp: number

export interface RealtimeHarness {
  readonly broker: RealtimeBroker
  readonly events: CollectedEvent[]
  /** Push raw frame al MockEventSource per `name` di canale (test SSE). */
  pushSseEvent(channelName: string, data: string, id?: string, eventType?: string): void
  /** Push raw frame al MockWebSocket per `name` di canale (test WS). */
  pushWsFrame(channelName: string, frame: string): void
  /** Simula __open() per il canale. */
  openChannel(channelName: string): void
  /** Simula __close() server-side per il canale. */
  closeChannel(channelName: string, code?: number, reason?: string): void
  /** Simula __error() server-side. */
  errorChannel(channelName: string): void
  /** Reset completo: clear events, restore globals, drop broker. */
  reset(): void

/**
 * Crea harness F4 con globals patch su EventSource + WebSocket.
 *
 * @example
 * ```ts
 * const h = createRealtimeHarness({ realtime: { channels: [{ name: 'orders', url: 'https://x' }] } })
 * h.openChannel('orders')
 * h.pushSseEvent('orders', '{"city":"Roma"}', 'evt-1', 'orders.update')
 * expect(h.events.find(e => e.topic === 'orders.update')).toBeDefined()
 * h.reset()
 * ```
 */
export function createRealtimeHarness(opts: RealtimeHarnessOptions = {}): RealtimeHarness {
  // Patch globals: EventSource + WebSocket → mock.
  const originalES = (globalThis as { EventSource?: typeof EventSource }).EventSource
  const originalWS = (globalThis as { WebSocket?: typeof WebSocket }).WebSocket
  ;(globalThis as { EventSource?: typeof EventSource }).EventSource = MockEventSource as unknown as typeof EventSource
  ;(globalThis as { WebSocket?: typeof WebSocket }).WebSocket = MockWebSocket as unknown as typeof WebSocket

  MockEventSource.__reset()
  MockWebSocket.__reset()

  const events: CollectedEvent[] = []
  const broker = createRealtimeBroker(opts)

  // W-3 fix — niente monkey-patch di broker.publish (mutava l'oggetto pubblico e mancava
  // il path interno `manager.publishFn → broker.inner.publish` ORIGINALE).
  // Pattern corretto: subscribe wildcard '*' (CORE-08 supporta wildcard subscribe per F1).
  // Cattura tutti gli eventi sia interni (system.realtime.*) sia di canale (orders.created etc.)
  // mantenendo l'invariante della pipeline §28 step 1-13.
  broker.subscribe('*', (ev: { topic: string; payload: unknown; source?: unknown; id?: string }) => {
    events.push({
      topic: ev.topic,
      payload: ev.payload,
      timestamp: Date.now(),
      ...(ev.source !== undefined && { source: ev.source as never }),
      ...(ev.id !== undefined && { id: ev.id }),
    })
  })

  // Convention: ogni canale di test deve avere URL nella forma `?_channel=<name>` (la harness
  // inserisce automaticamente questo query param in un buildUrl wrapper se la def non lo contiene).

  // registrano in `byChannelName` (vedi mock files plan 04-05/06).
  function findChannel(name: string): { es?: MockEventSource; ws?: MockWebSocket } {
    // B-NEW-2 fix iter 2 — `byChannelName` è owned da 04-05/04-06 come property statica
    // permanente; nessuna optional chain. Se undefined il lookup ritorna undefined e i
    // metodi pushSseEvent/pushWsFrame/openChannel/closeChannel lanciano un errore esplicito
    // (intentional — indica test setup error: URL del canale manca `?_channel=<name>`).
    const es = MockEventSource.byChannelName.get(name)
    const ws = MockWebSocket.byChannelName.get(name)
    return { ...(es && { es }), ...(ws && { ws }) }
  }

  return {
    broker,
    events,
    // B-2 fix — strict channel routing via byChannelName Map. NIENTE fallback a "ultimo instance"
    // (causava cross-canale pollution). Il name DEVE matchare un canale registrato con
    // URL contenente `?_channel=<name>` o `&_channel=<name>`.
    pushSseEvent(channelName, data, id, eventType = 'message') {
      const target = findChannel(channelName).es
      if (!target) throw new Error(`No SSE channel found for name='${channelName}' — verify URL has _channel=${channelName}`)
      target.__message(data, id, eventType)
    },
    pushWsFrame(channelName, frame) {
      const target = findChannel(channelName).ws
      if (!target) throw new Error(`No WS channel found for name='${channelName}' — verify URL has _channel=${channelName}`)
      target.__message(frame)
    },
    openChannel(channelName) {
      const ch = findChannel(channelName)
      ch.es?.__open()
      ch.ws?.__open()
      if (!ch.es && !ch.ws) throw new Error(`No channel found for name='${channelName}'`)
    },
    closeChannel(channelName, code = 1006, reason = 'abnormal') {
      const ch = findChannel(channelName)
      ch.ws?.__close(code, reason)
      ch.es?.__error()
    },
    errorChannel(channelName) {
      const ch = findChannel(channelName)
      ch.es?.__error()
      ch.ws?.__error()
    },
    reset() {
      events.length = 0
      MockEventSource.__reset()
      MockWebSocket.__reset()
      ;(globalThis as { EventSource?: typeof EventSource }).EventSource = originalES
      ;(globalThis as { WebSocket?: typeof WebSocket }).WebSocket = originalWS
      vi.restoreAllMocks()
    },
  }
}
