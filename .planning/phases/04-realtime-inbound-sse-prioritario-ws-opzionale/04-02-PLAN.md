```typescript
/**
 * Envelope JSON default per WebSocket frames (D-106).
 *
 * Contract:
 * - `topic` → BrokerEvent.topic. NON vuota.
 * - `data` → payload raw (poi normalizzato dal mapper step 4 §28 in plan 04-08 via RouterBroker).
 * - `id` opzionale → BrokerEvent.id (se assente, generato via nanoid dall'adapter).
 *
 * Frame non-conformi (parse fail o missing topic) → l'adapter (plan 04-06) publish
 * `network.error` con `category: 'protocol'` + descarta il frame (D-106 + Q2 RESEARCH).
 */
export interface FrameEnvelope {
  readonly topic: string
  readonly data: unknown
  readonly id?: string
}

/**
 * Result discriminato del parser (pattern simile a `RouteOutcome` F3).
 *
 * `ok: true` → envelope estratto correttamente.
 * `ok: false` → ragione enumerata + `raw` per debug/forensics.
 */
export type FrameParseResult =
  | { readonly ok: true; readonly envelope: FrameEnvelope }
  | {
      readonly ok: false
      readonly reason: 'malformed-json' | 'missing-topic' | 'invalid-shape'
      readonly raw: string
    }
```

**Step 3 (GREEN) — Crea `packages/gateway/src/sse-ws/frame-parser.ts`** copiando da PATTERNS.md §2.5 (verbatim, deterministico):

```typescript
import type { FrameEnvelope, FrameParseResult } from './types/frame-envelope'

/**
 * Topic riservati internal al protocollo realtime (D-111).
 *
 * Filtrati dall'adapter WS PRIMA del publish (PITFALL §11.7 — match strict, no prefix
 * wildcard). Topic utente come `weather.__ping__` (raro ma legittimo) passano attraverso.
 */
export const INTERNAL_TOPICS = Object.freeze({
  PING: '__ping__',
  PONG: '__pong__',
} as const)

/**
 * Parse un frame WebSocket testuale come `FrameEnvelope` JSON (D-106).
 *
 * Contract:
 * - `raw` deve essere stringa JSON con shape `{ topic: string, data: unknown, id?: string }`.
 * - Frame non-conformi → `{ ok: false, reason }` (caller publish `network.error`
 *   con `category: 'protocol'` per Q2 / RESEARCH §7.1).
 * - NIENTE throw: il caller (websocket-adapter.ts plan 04-06) gestisce graceful via `.ok`.
 *
 * Pattern identico a `parseRetryAfter` (gateway/src/http/retry-after-parser.ts):
 * narrow input/output, no side-effect, no throw, test deterministici tier-1 jsdom.
 *
 * @param raw - Stringa raw del frame (da `MessageEvent.data` su `'message'` event).
 * @returns FrameParseResult discriminato — caller usa `.ok` per branching.
 *
 * @example
 * ```ts
 * const result = parseFrame(ev.data)
 * if (!result.ok) {
 *   publishFn({ topic: 'network.error', payload: { reason: result.reason } })
 *   return
 * }
 * publishFn({ topic: result.envelope.topic, payload: result.envelope.data })
 * ```
 */
export function parseFrame(raw: unknown): FrameParseResult {
  if (typeof raw !== 'string') {
    return { ok: false, reason: 'malformed-json', raw: String(raw) }
  }
  let parsed: unknown
  try {
    parsed = JSON.parse(raw)
  } catch {
    return { ok: false, reason: 'malformed-json', raw }
  }
  // null è typeof 'object' in JS — esplicito check.
  if (parsed === null || typeof parsed !== 'object' || Array.isArray(parsed)) {
    return { ok: false, reason: 'invalid-shape', raw }
  }
  const obj = parsed as Record<string, unknown>
  if (typeof obj.topic !== 'string' || obj.topic.length === 0) {
    return { ok: false, reason: 'missing-topic', raw }
  }
  // Build envelope con `id` solo se presente E string (Test 10 — id non-string ignorato).
  const envelope: FrameEnvelope =
    typeof obj.id === 'string'
      ? { topic: obj.topic, data: obj.data, id: obj.id }
      : { topic: obj.topic, data: obj.data }
  return { ok: true, envelope }
}

/**
 * Verifica se un topic è riservato internamente al protocollo realtime
 * (D-111 — `__ping__`/`__pong__` per heartbeat WS).
 *
 * **PITFALL §11.7 (chiave) — strict match, NON prefix:**
 * Topic legittimi consumer come `weather.__ping__` NON devono essere filtrati. Il
 * confronto è `topic === '__ping__' || topic === '__pong__'` esatto.
 *
 * @example
 * ```ts
 * isInternalTopic('__ping__')         // true (filter)
 * isInternalTopic('__pong__')         // true (filter)
 * isInternalTopic('weather.__ping__') // false (passa attraverso)
 * ```
 */
export function isInternalTopic(topic: string): boolean {
  return topic === INTERNAL_TOPICS.PING || topic === INTERNAL_TOPICS.PONG
}
```
