Pattern factory closure + Map (analog circuit-breaker.ts):

```ts
import type { TaskState, WorkerTaskOutcome } from './types'

interface TrackerState {
  state: TaskState
  startedAt: number
  correlationId: string
  result?: unknown
  errorCode?: string
  errorMessage?: string
  endedAt?: number
}

export interface TaskTrackerSnapshot {
  readonly tasksActive: number
  readonly tasksCompleted: number
  readonly lateResponses: number
  readonly tasks: readonly {
    readonly taskId: string
    readonly state: TaskState
    readonly correlationId: string
    readonly startedAt: number
  }[]
}

export interface TaskTracker {
  register(taskId: string, correlationId: string): void
  markDone(taskId: string, result: unknown): boolean
  markTimeout(taskId: string): boolean
  markCancelled(taskId: string): boolean
  markError(taskId: string, errorCode: string, errorMessage: string): boolean
  getOutcome(taskId: string): WorkerTaskOutcome | undefined
  getDebugSnapshot(): TaskTrackerSnapshot
}

/**
 * F5 task tracker — state machine atomico per Pitfall 2C closure (D-133).
 *
 * **Atomicità implicita**: JS event loop single-threaded — `tasks.get(taskId)` + check
 * `state === 'pending'` + `state = target` è una sequenza atomica per lo standard JS
 * (no preemption mid-statement). Niente lock necessari.
 *
 * **Counter lateResponses** (Claude's Discretion lockata in CONTEXT): incrementato quando
 * una transition arriva su un task non-pending o mai registrato. Esposto in
 * `getDebugSnapshot()` per audit retroattivo.
 *
 * @see D-133, D-134, D-151 #3
 */
export function createTaskTracker(): TaskTracker {
  const tasks = new Map<string, TrackerState>()
  let tasksCompleted = 0
  let lateResponses = 0

  function tryTransition(
    taskId: string,
    target: Exclude<TaskState, 'pending'>,
    update?: Partial<TrackerState>,
  ): boolean {
    const s = tasks.get(taskId)
    if (s === undefined || s.state !== 'pending') {
      lateResponses++
      return false
    }
    s.state = target
    s.endedAt = Date.now()
    if (update !== undefined) Object.assign(s, update)
    tasksCompleted++
    return true
  }

  return {
    register(taskId, correlationId) {
      tasks.set(taskId, {
        state: 'pending',
        startedAt: Date.now(),
        correlationId,
      })
    },
    markDone(taskId, result) {
      return tryTransition(taskId, 'done', { result })
    },
    markTimeout(taskId) {
      return tryTransition(taskId, 'timeout')
    },
    markCancelled(taskId) {
      return tryTransition(taskId, 'cancelled')
    },
    markError(taskId, errorCode, errorMessage) {
      return tryTransition(taskId, 'error', { errorCode, errorMessage })
    },
    getOutcome(taskId) {
      const s = tasks.get(taskId)
      if (s === undefined) return undefined
      return {
        taskId,
        correlationId: s.correlationId,
        state: s.state,
        result: s.result,
        errorCode: s.errorCode,
        errorMessage: s.errorMessage,
        elapsedMs: (s.endedAt ?? Date.now()) - s.startedAt,
      }
    },
    getDebugSnapshot() {
      let active = 0
      const taskList: TaskTrackerSnapshot['tasks'][number][] = []
      for (const [taskId, s] of tasks) {
        if (s.state === 'pending') active++
        taskList.push({
          taskId,
          state: s.state,
          correlationId: s.correlationId,
          startedAt: s.startedAt,
        })
      }
      return { tasksActive: active, tasksCompleted, lateResponses, tasks: taskList }
    },
  }
}
```
