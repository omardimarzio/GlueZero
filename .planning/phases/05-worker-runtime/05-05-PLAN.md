import { createBrokerError } from '@sembridge/core'
import type { WorkerDescriptor } from './types'

export interface WorkerEntry {
  readonly desc: WorkerDescriptor
  readonly ownerId: string
  readonly registeredAt: number
}

export interface WorkerRegistrySnapshot {
  readonly workerCount: number
  readonly byOwner: Readonly<Record<string, number>>
}

const MAX_POOL_SIZE_HARD = 8

export class WorkerRegistry {
  private readonly entries = new Map<string, WorkerEntry>()

  register(desc: WorkerDescriptor, ownerId: string): void {
    this.validateDescriptor(desc)
    if (this.entries.has(desc.id)) {
      throw createBrokerError({
        code: 'worker.id.duplicate',
        message: `Worker id '${desc.id}' already registered (owner: ${this.entries.get(desc.id)?.ownerId}). Use unique id or unregister first.`,
        category: 'config',
        details: { workerId: desc.id, existingOwner: this.entries.get(desc.id)?.ownerId, requestedOwner: ownerId },
      })
    }
    this.entries.set(desc.id, { desc, ownerId, registeredAt: Date.now() })
  }

  get(id: string): WorkerEntry | undefined {
    return this.entries.get(id)
  }

  validateTask(workerId: string, taskName: string): boolean {
    const entry = this.entries.get(workerId)
    if (entry === undefined) {
      throw createBrokerError({
        code: 'worker.unknown',
        message: `Worker id '${workerId}' is not registered.`,
        category: 'config',
        details: { workerId },
      })
    }
    return entry.desc.tasks.includes(taskName)
  }

  listByOwner(ownerId: string): WorkerEntry[] {
    const out: WorkerEntry[] = []
    for (const entry of this.entries.values()) {
      if (entry.ownerId === ownerId) out.push(entry)
    }
    return out
  }

  unregister(id: string): boolean {
    return this.entries.delete(id)
  }

  /** D-126 cascade — analog F4 disconnectByOwner. */
  unregisterByOwner(ownerId: string): readonly string[] {
    const removed: string[] = []
    for (const [id, entry] of this.entries) {
      if (entry.ownerId === ownerId) {
        this.entries.delete(id)
        removed.push(id)
      }
    }
    return removed
  }

  getDebugSnapshot(): WorkerRegistrySnapshot {
    const byOwner: Record<string, number> = {}
    for (const entry of this.entries.values()) {
      byOwner[entry.ownerId] = (byOwner[entry.ownerId] ?? 0) + 1
    }
    return { workerCount: this.entries.size, byOwner }
  }

  private validateDescriptor(desc: WorkerDescriptor): void {
    if (desc.id === '' || desc.id === undefined) {
      throw createBrokerError({
        code: 'worker.descriptor.invalid',
        message: `WorkerDescriptor.id is required and must be non-empty`,
        category: 'config',
      })
    }
    if (!Array.isArray(desc.tasks) || desc.tasks.length === 0) {
      throw createBrokerError({
        code: 'worker.descriptor.invalid',
        message: `WorkerDescriptor '${desc.id}'.tasks must be non-empty readonly array (D-124 fail-fast)`,
        category: 'config',
        details: { workerId: desc.id, tasks: desc.tasks as readonly string[] },
      })
    }
    if (typeof desc.factory !== 'function') {
      throw createBrokerError({
        code: 'worker.descriptor.invalid',
        message: `WorkerDescriptor '${desc.id}'.factory must be a function (D-123)`,
        category: 'config',
      })
    }
    // D-128 pool size validation
    if (desc.mode === 'pool' && desc.size !== undefined && desc.size > MAX_POOL_SIZE_HARD && desc.allowUnboundedPool !== true) {
      throw createBrokerError({
        code: 'worker.pool.size.exceeded',
        message: `Pool size ${desc.size} exceeds hard cap ${MAX_POOL_SIZE_HARD}. Set allowUnboundedPool: true to bypass (with console.warn).`,
        category: 'config',
        details: { workerId: desc.id, requestedSize: desc.size, maxSize: MAX_POOL_SIZE_HARD },
      })
    }
  }
}

import { createBackpressureStrategy, type BackpressureStrategy, type BackpressurePolicyConfig } from '@sembridge/gateway/http'
import { createBrokerError } from '@sembridge/core'
import { WorkerBridge } from './worker-bridge'
import type { WorkerRegistry } from './worker-registry'
import type { WorkerDescriptor, ProgressPayload } from './types'

export function defaultPoolSize(): number {
  const hwc = (typeof navigator !== 'undefined' && typeof navigator.hardwareConcurrency === 'number')
    ? navigator.hardwareConcurrency
    : 4
  return Math.min(hwc, 4)
}

export interface WorkerPoolDeps {
  readonly registry: WorkerRegistry
  readonly WorkerCtor?: typeof Worker
  /** D-130 — opt-in custom backpressure strategy; default queue-bounded maxSize=1000. */
  readonly backpressure?: BackpressureStrategy
  /** D-139 — assertSerializable mode override forwarded to bridges. */
  readonly assertSerializableMode?: 'always' | 'dev' | 'off'
}

interface PoolSlot {
  readonly bridge: WorkerBridge
  busy: boolean
  currentTaskId?: string
}

export interface WorkerPoolSnapshot {
  readonly activeBridges: number
  readonly queuedTasks: number
  readonly byWorkerId: Readonly<Record<string, { readonly spawned: number; readonly busy: number }>>
}

const DEFAULT_BACKPRESSURE: BackpressurePolicyConfig = {
  policy: 'queue-bounded',
  maxSize: 1000,
}

export class WorkerPool {
  private readonly slotsByWorker = new Map<string, PoolSlot[]>()
  private readonly backpressure: BackpressureStrategy
  private warnedUnbounded = false

  constructor(private readonly deps: WorkerPoolDeps) {
    this.backpressure = deps.backpressure ?? createBackpressureStrategy({
      defaultPolicy: DEFAULT_BACKPRESSURE,
    })
  }

  /**
   * D-130 — schedule task via F3 BackpressureStrategy (critical bypass).
   * routeId acts as backpressure key.
   */
  async schedule<T>(
    routeId: string,
    priority: 'low' | 'normal' | 'high' | 'critical',
    task: () => Promise<T>,
  ): Promise<T> {
    if (priority === 'critical') {
      return await task()  // D-130 critical bypass (Pitfall 4.C)
    }
    return await this.backpressure.schedule(routeId, priority, task)
  }

  async dispatchOnSlot(
    workerId: string,
    taskName: string,
    payload: unknown,
    signal: AbortSignal,
    onProgress?: (p: ProgressPayload) => void,
    transferable?: readonly string[],
  ): Promise<unknown> {
    const slot = await this.acquireSlot(workerId)
    try {
      return await slot.bridge.dispatch(taskName, payload, signal, onProgress, { transferable })
    } finally {
      this.releaseSlot(workerId, slot)
    }
  }

  /** D-129 lazy first dispatch — spawn slot on demand up to size. */
  async acquireSlot(workerId: string): Promise<PoolSlot> {
    const entry = this.deps.registry.get(workerId)
    if (entry === undefined) {
      throw createBrokerError({
        code: 'worker.unknown',
        message: `Worker '${workerId}' is not registered`,
        category: 'config',
      })
    }
    const desc = entry.desc
    const targetSize = this.resolveSize(desc)
    let slots = this.slotsByWorker.get(workerId)
    if (slots === undefined) {
      slots = []
      this.slotsByWorker.set(workerId, slots)
    }
    // Cerca slot libero
    for (const s of slots) {
      if (!s.busy) {
        s.busy = true
        return s
      }
    }
    // Espandi pool se sotto capacity
    if (slots.length < targetSize) {
      const bridge = new WorkerBridge(desc, {
        WorkerCtor: this.deps.WorkerCtor,
        assertSerializableMode: this.deps.assertSerializableMode,
      })
      const newSlot: PoolSlot = { bridge, busy: true }
      slots.push(newSlot)
      return newSlot
    }
    // Pool saturo — caller dovrebbe usare schedule() per backpressure;
    // qui blocca con polling Promise (semplice, no sophisticated wait queue per V1)
    return await this.waitForFreeSlot(workerId, slots)
  }

  releaseSlot(workerId: string, slot: PoolSlot): void {
    slot.busy = false
    slot.currentTaskId = undefined
  }

  /** D-131 fallback — bridge fault recovery. */
  respawn(workerId: string, slotIndex: number): void {
    const slots = this.slotsByWorker.get(workerId)
    if (slots === undefined || slotIndex >= slots.length) return
    const oldSlot = slots[slotIndex]
    if (oldSlot === undefined) return
    try { oldSlot.bridge.terminate() } catch { /* idempotent */ }
    const entry = this.deps.registry.get(workerId)
    if (entry === undefined) return
    const newBridge = new WorkerBridge(entry.desc, {
      WorkerCtor: this.deps.WorkerCtor,
      assertSerializableMode: this.deps.assertSerializableMode,
    })
    slots[slotIndex] = { bridge: newBridge, busy: false }
  }

  /** D-126 ext F5 cascade — terminate tutti i worker owned + svuota queue. */
  terminateByOwner(ownerId: string): void {
    const owned = this.deps.registry.listByOwner(ownerId)
    for (const entry of owned) {
      const slots = this.slotsByWorker.get(entry.desc.id)
      if (slots !== undefined) {
        for (const s of slots) {
          try { s.bridge.terminate() } catch { /* idempotent */ }
        }
        this.slotsByWorker.delete(entry.desc.id)
      }
    }
    // F3 backpressure drain non è per-owner — il consumer (worker-handler 05-06)
    // gestirà drain per routeId al unregisterPlugin
  }

  /** Test util — eager spawn per assertions deterministiche. */
  spawnEager(workerId: string): void {
    const entry = this.deps.registry.get(workerId)
    if (entry === undefined) return
    const desc = entry.desc
    const bridge = new WorkerBridge(desc, {
      WorkerCtor: this.deps.WorkerCtor,
      assertSerializableMode: this.deps.assertSerializableMode,
    })
    // Force factory invocation per Test 6
    void bridge.dispatch(desc.tasks[0] ?? 'noop', {}, new AbortController().signal).catch(() => { /* ignore */ })
    let slots = this.slotsByWorker.get(workerId)
    if (slots === undefined) {
      slots = []
      this.slotsByWorker.set(workerId, slots)
    }
    slots.push({ bridge, busy: false })
  }

  getDebugSnapshot(): WorkerPoolSnapshot {
    let active = 0
    const byWorkerId: Record<string, { spawned: number; busy: number }> = {}
    for (const [id, slots] of this.slotsByWorker) {
      const busy = slots.filter((s) => s.busy).length
      byWorkerId[id] = { spawned: slots.length, busy }
      active += slots.length
    }
    return { activeBridges: active, queuedTasks: 0 /* TODO inspect backpressure queue */, byWorkerId }
  }

  private resolveSize(desc: WorkerDescriptor): number {
    if (desc.mode === 'dedicated') return 1
    const requested = desc.size ?? defaultPoolSize()
    if (requested > 8 && desc.allowUnboundedPool !== true) {
      // Already validated at register; defensive cap
      return 8
    }
    if (requested > 8 && desc.allowUnboundedPool === true && !this.warnedUnbounded) {
      // Pitfall 7.D protection — warn once per worker
      // eslint-disable-next-line no-console
      console.warn(`[SemBridge] WorkerPool '${desc.id}' configured with allowUnboundedPool size=${requested}. Risk of pool storm — monitor RAM usage.`)
      this.warnedUnbounded = true
    }
    return requested
  }

  private async waitForFreeSlot(_workerId: string, slots: PoolSlot[]): Promise<PoolSlot> {
    // Polling implementazione minimal V1 — handler 05-06 dovrebbe routare via schedule()
    // per backpressure. Questo è defensive fallback.
    return new Promise((resolve) => {
      const tick = (): void => {
        for (const s of slots) {
          if (!s.busy) {
            s.busy = true
            resolve(s)
            return
          }
        }
        setTimeout(tick, 5)
      }
      tick()
    })
  }
}
