```ts
import { nanoid } from 'nanoid'
import { createBrokerError, type BrokerError, type BrokerEvent } from '@sembridge/core'
import type { WorkerRegistry } from './worker-registry'
import type { WorkerPool } from './worker-pool'
import type { TaskTracker } from './task-tracker'
import type { RouteWorkerDefinition, ProgressPayload } from './types'

export interface WorkerHandlerDeps {
  readonly registry: WorkerRegistry
  readonly pool: WorkerPool
  readonly tracker: TaskTracker
  readonly publishFn: (
    topic: string,
    payload: unknown,
    options?: { source?: { type: string; id: string; name?: string }; correlationId?: string; priority?: 'low' | 'normal' | 'high' | 'critical' },
  ) => void | Promise<void>
}

export interface WorkerHandler {
  execute(
    event: BrokerEvent,
    route: RouteWorkerDefinition,
    signal: AbortSignal,
  ): Promise<void>
}

const DEFAULT_TIMEOUT_MS = 30_000
const DEFAULT_CONCURRENCY: 'serial' | 'parallel' | 'latest-only' = 'latest-only'

/**
 * F5 createWorkerHandler — Strategy che orchestra registry → pool → bridge → tracker
 * → publish outcome (D-152 step 9 dispatch + D-153 mapping canonical → output).
 *
 * **Opzione B research §7.2:** Questo handler viene invocato dal `WorkerBroker.publish`
 * intercept PRE-delegate a `inner.publish` — evita modifica `packages/routing/` (D-83 strict).
 *
 * **Topic naming (D-146):** auto-derive con suffix-replace su `<topic>.requested`:
 * - success → `<entity>.<action>.completed`
 * - progress → `<entity>.<action>.progress`
 * - error → `<entity>.<action>.failed`
 * - override esplicito via `route.publishes.{success|progress|error}`
 *
 * **State machine atomico (D-133 closure):** tracker.markDone/Timeout/Cancelled/Error
 * ritornano boolean — late responses scartate silenziosamente (Pitfall 2C).
 *
 * **correlationId end-to-end (D-134):** ogni outcome event propaga
 * `event.correlationId ?? event.id`.
 *
 * @see RESEARCH §7.2 (Opzione B rationale)
 */
export function createWorkerHandler(deps: WorkerHandlerDeps): WorkerHandler {
  return {
    async execute(event, route, externalSignal) {
      const correlationId = (event as BrokerEvent & { correlationId?: string }).correlationId ?? event.id
      const taskId = nanoid()

      // Validate worker + task (D-124 fail-fast — runtime safety oltre register-time)
      const entry = deps.registry.get(route.worker)
      if (entry === undefined) {
        await publishFailure(deps.publishFn, route, event, correlationId, createBrokerError({
          code: 'worker.unknown',
          message: `Worker '${route.worker}' is not registered`,
          category: 'config',
          details: { workerId: route.worker, routeId: route.id, topic: event.topic },
        }))
        return
      }
      try {
        if (!deps.registry.validateTask(route.worker, route.task)) {
          throw createBrokerError({
            code: 'worker.task.unknown',
            message: `Worker '${route.worker}' does not declare task '${route.task}'`,
            category: 'config',
            details: { workerId: route.worker, taskName: route.task, routeId: route.id, topic: event.topic },
          })
        }
      } catch (err) {
        await publishFailure(deps.publishFn, route, event, correlationId, asBrokerError(err))
        return
      }

      // D-145 default timeout
      const timeoutMs = route.policies?.timeout ?? DEFAULT_TIMEOUT_MS
      const concurrency = route.policies?.concurrency ?? DEFAULT_CONCURRENCY

      // D-134 register tracker
      deps.tracker.register(taskId, correlationId)

      // Combined signal: external + timeout + concurrency abort
      const internalCtrl = new AbortController()
      const onExternalAbort = (): void => internalCtrl.abort('external')
      externalSignal.addEventListener('abort', onExternalAbort, { once: true })
      const timeoutHandle = setTimeout(() => {
        if (deps.tracker.markTimeout(taskId)) {
          internalCtrl.abort('timeout')
        }
      }, timeoutMs)

      const onProgress = (p: ProgressPayload): void => {
        // D-138 progress passa per pipeline canonical → consumer
        const progressTopic = route.publishes?.progress ?? deriveTopic(event.topic, 'progress')
        void deps.publishFn(progressTopic, p, {
          source: { type: 'worker', id: route.worker, name: route.task },
          correlationId,
        })
      }

      try {
        // D-130 pool.schedule (BackpressureStrategy F3 critical bypass)
        const priority = (event as BrokerEvent & { priority?: 'low' | 'normal' | 'high' | 'critical' }).priority ?? 'normal'
        const result = await deps.pool.schedule(route.id, priority, async () => {
          return await deps.pool.dispatchOnSlot(
            route.worker,
            route.task,
            event.payload,
            internalCtrl.signal,
            onProgress,
            route.transferable,
          )
        })
        // D-133 atomic CAS — solo se ancora pending pubblica completed
        if (deps.tracker.markDone(taskId, result)) {
          const successTopic = route.publishes?.success ?? deriveTopic(event.topic, 'completed')
          await deps.publishFn(successTopic, result, {
            source: { type: 'worker', id: route.worker, name: route.task },
            correlationId,
          })
        }
        // else: late response — scartata silenziosamente (Pitfall 2C closure)
      } catch (err) {
        // D-133 atomic CAS — solo se ancora pending pubblica failed
        const reason = internalCtrl.signal.reason
        if (reason === 'timeout') {
          // markTimeout già chiamato — pubblica failed con timeout shape
          await publishFailure(deps.publishFn, route, event, correlationId, createBrokerError({
            code: 'worker.timeout',
            message: `Worker '${route.worker}' task '${route.task}' exceeded timeout ${timeoutMs}ms`,
            category: 'worker',
            details: { workerId: route.worker, taskName: route.task, routeId: route.id, topic: event.topic, timeoutMs },
          }))
        } else if (reason === 'external') {
          if (deps.tracker.markCancelled(taskId)) {
            await publishFailure(deps.publishFn, route, event, correlationId, createBrokerError({
              code: 'worker.cancelled',
              message: `Worker '${route.worker}' task '${route.task}' cancelled by signal`,
              category: 'worker',
              details: { workerId: route.worker, taskName: route.task, routeId: route.id, topic: event.topic },
            }))
          }
        } else {
          // Generic error
          if (deps.tracker.markError(taskId, asBrokerError(err).code, asBrokerError(err).message)) {
            await publishFailure(deps.publishFn, route, event, correlationId, asBrokerError(err))
          }
        }
      } finally {
        clearTimeout(timeoutHandle)
        externalSignal.removeEventListener('abort', onExternalAbort)
      }
    },
  }
}

function asBrokerError(err: unknown): BrokerError {
  if (typeof err === 'object' && err !== null && 'code' in err && 'category' in err && 'message' in err) {
    return err as BrokerError
  }
  return createBrokerError({
    code: 'worker.error',
    message: err instanceof Error ? err.message : String(err),
    category: 'worker',
  })
}

async function publishFailure(
  publishFn: WorkerHandlerDeps['publishFn'],
  route: RouteWorkerDefinition,
  event: BrokerEvent,
  correlationId: string,
  error: BrokerError,
): Promise<void> {
  const errorTopic = route.publishes?.error ?? deriveTopic(event.topic, 'failed')
  // S8 Sanitization (T-03-07-01 carryover) — solo code/category/message + safe details
  const sanitized = {
    code: error.code,
    category: error.category,
    message: error.message,
    routeId: route.id,
    topic: event.topic,
    eventId: event.id,
    workerId: route.worker,
    taskName: route.task,
  }
  await publishFn(errorTopic, sanitized, {
    source: { type: 'worker', id: route.worker, name: route.task },
    correlationId,
  })
  // ERR-02 ext F5 — emit also worker.error topic (D-140)
  await publishFn('worker.error', sanitized, {
    source: { type: 'worker', id: route.worker, name: route.task },
    correlationId,
  })
}

/** D-146 topic auto-derive — analog F3 outcome-collector.ts:24-28 suffix-replace logic. */
export function deriveTopic(sourceTopic: string, suffix: 'completed' | 'progress' | 'failed'): string {
  if (sourceTopic.endsWith('.requested')) {
    return sourceTopic.slice(0, -'.requested'.length) + '.' + suffix
  }
  return sourceTopic + '.' + suffix
}
```

```ts
import { RouterBroker, type RouterBrokerConfig, type RouterPublishOptions } from '@sembridge/routing'
import type { PluginDescriptor, BrokerEvent, Subscription, BrokerLogLevel } from '@sembridge/core'
import { createBrokerError } from '@sembridge/core'
import { WorkerRegistry } from './worker-registry'
import { WorkerPool } from './worker-pool'
import { createTaskTracker, type TaskTracker } from './task-tracker'
import { createWorkerHandler, type WorkerHandler } from './worker-handler'
import type { RouteWorkerDefinition, WorkerConfig } from './types'

export interface WorkerBrokerConfig extends RouterBrokerConfig {
  /** F5 augment — D-122 (declaration merging in augment.ts already merges this with BrokerConfig). */
  readonly workers?: WorkerConfig
  /** Worker routes registrate al boot (alternativa a registerWorkerRoute runtime). */
  readonly workerRoutes?: readonly RouteWorkerDefinition[]
  /** WorkerCtor DI per Tier-1 jsdom test (default `globalThis.Worker`). */
  readonly WorkerCtor?: typeof Worker
}

/**
 * F5 WorkerBroker — composition wrapper di RouterBroker (D-121, D-83 strict carryover).
 *
 * **Opzione B research §7.2 — D-83 strict preservation:**
 * Il `publish(topic)` override intercetta topic matching una worker route registrata
 * prima di delegare a `inner.publish` (RouterBroker F3). In questo modo F5:
 * - NON modifica `packages/routing/route-resolver.ts` né `route-executor.ts`
 * - NON viola D-83 (zero diff packages/routing/)
 * - Riusa pipeline §28 mapper F2 step 5-6 + step 11-12 invariati per il flusso non-worker
 * - Aggiunge step 9 dispatch worker pre-publish per route worker matching
 *
 * Per le route NON-worker (HTTP, local, cache, composite), il publish è delegato
 * trasparentemente a `inner.publish` — pipeline F3 normale.
 *
 * **Cascade D-126 + LIFE-02 ext F5:**
 * - `registerPlugin(desc)` → `inner.registerPlugin` + auto-registra `desc.workers` con
 *   `ownerId=desc.id` nel registry (con `system.warn` su register failure)
 * - `unregisterPlugin(id)` → `inner.unregisterPlugin` + `registry.unregisterByOwner` +
 *   `pool.terminateByOwner` (3-step cascade, try/catch isolato per idempotency)
 *
 * @see .planning/phases/05-worker-runtime/05-RESEARCH.md §7.2
 * @see D-121, D-122, D-126, D-152
 */
export class WorkerBroker {
  private readonly inner: RouterBroker
  private readonly registry: WorkerRegistry
  private readonly pool: WorkerPool
  private readonly tracker: TaskTracker
  private readonly handler: WorkerHandler
  private readonly workerRoutes = new Map<string, RouteWorkerDefinition>() // topic → route
  private readonly externalAbortControllers = new Map<string, AbortController>()

  constructor(config: WorkerBrokerConfig = {}) {
    this.inner = new RouterBroker(config)
    this.registry = new WorkerRegistry()
    this.pool = new WorkerPool({
      registry: this.registry,
      WorkerCtor: config.WorkerCtor,
      assertSerializableMode: config.workers?.assertSerializable,
    })
    this.tracker = createTaskTracker()
    this.handler = createWorkerHandler({
      registry: this.registry,
      pool: this.pool,
      tracker: this.tracker,
      publishFn: (topic, payload, opts) => this.inner.publish(topic, payload as never, opts as RouterPublishOptions),
    })

    // Bootstrap worker routes from config
    if (config.workerRoutes) {
      for (const route of config.workerRoutes) {
        this.registerWorkerRoute(route)
      }
    }
  }

  // ---------- Public API ----------

  /**
   * Opzione B (RESEARCH §7.2) — publish intercept pre-delegate.
   * Topic matching worker route → execute via handler; altrimenti delegate inner.
   */
  async publish(topic: string, payload: unknown, options?: RouterPublishOptions): Promise<void> {
    const workerRoute = this.workerRoutes.get(topic)
    if (workerRoute === undefined) {
      return await this.inner.publish(topic, payload, options)
    }
    // Costruisci BrokerEvent canonico per handler
    const event: BrokerEvent = {
      id: options?.id ?? `evt-${Date.now()}-${Math.random().toString(36).slice(2)}`,
      topic,
      payload: payload as never,
      timestamp: Date.now(),
      source: options?.source ?? { type: 'system', id: 'worker-broker' },
    } as BrokerEvent

    // External AbortController per scope publish (consumer può cancel via signal opt-in)
    const abortKey = `${topic}:${event.id}`
    const ctrl = new AbortController()
    this.externalAbortControllers.set(abortKey, ctrl)
    try {
      await this.handler.execute(event, workerRoute, ctrl.signal)
    } finally {
      this.externalAbortControllers.delete(abortKey)
    }
  }

  subscribe(topic: string, handler: (event: BrokerEvent) => void | Promise<void>, options?: { signal?: AbortSignal }): Subscription {
    return this.inner.subscribe(topic, handler, options)
  }

  async registerPlugin(descriptor: PluginDescriptor): Promise<void> {
    await this.inner.registerPlugin(descriptor)
    if (descriptor.workers !== undefined && descriptor.workers.length > 0) {
      for (const desc of descriptor.workers) {
        try {
          this.registry.register(desc, descriptor.id)
        } catch (err) {
          // pattern F4 W-5: niente silent catch — emit system.warn con dettagli
          this.inner.publish('system.warn', {
            plugin: descriptor.id,
            worker: desc.id,
            reason: 'worker-register-failed',
            error: errorToSafeShape(err),
          } as never, { source: { type: 'system', id: 'worker-broker', name: 'register-plugin' } } as RouterPublishOptions)
            .catch(() => { /* defensive */ })
        }
      }
    }
  }

  async unregisterPlugin(id: string): Promise<void> {
    // 3-step cascade (S2 carryover) con try/catch isolato (idempotent)
    try { await this.inner.unregisterPlugin(id) } catch { /* pattern F3 silent */ }
    try { this.registry.unregisterByOwner(id) } catch { /* idempotent */ }
    try { this.pool.terminateByOwner(id) } catch { /* idempotent */ }
  }

  registerWorker(desc: Parameters<WorkerRegistry['register']>[0]): void {
    this.registry.register(desc, 'system')
  }

  unregisterWorker(id: string): void {
    const entry = this.registry.get(id)
    if (entry !== undefined) {
      this.pool.terminateByOwner(entry.ownerId)
    }
    this.registry.unregister(id)
  }

  registerWorkerRoute(route: RouteWorkerDefinition): void {
    if (this.workerRoutes.has(route.topic)) {
      throw createBrokerError({
        code: 'worker.route.duplicate',
        message: `WorkerRoute on topic '${route.topic}' already registered (id: ${this.workerRoutes.get(route.topic)?.id})`,
        category: 'config',
        details: { topic: route.topic, existingRouteId: this.workerRoutes.get(route.topic)?.id, requestedRouteId: route.id },
      })
    }
    // Validate worker exists + task declared (D-124)
    if (this.registry.get(route.worker) !== undefined) {
      if (!this.registry.validateTask(route.worker, route.task)) {
        throw createBrokerError({
          code: 'worker.task.unknown',
          message: `Worker '${route.worker}' does not declare task '${route.task}'`,
          category: 'config',
          details: { workerId: route.worker, taskName: route.task, routeId: route.id },
        })
      }
    }
    this.workerRoutes.set(route.topic, route)
  }

  getDebugSnapshot(): {
    readonly registry: ReturnType<WorkerRegistry['getDebugSnapshot']>
    readonly pool: ReturnType<WorkerPool['getDebugSnapshot']>
    readonly tracker: ReturnType<TaskTracker['getDebugSnapshot']>
    readonly workerRoutes: number
  } {
    return {
      registry: this.registry.getDebugSnapshot(),
      pool: this.pool.getDebugSnapshot(),
      tracker: this.tracker.getDebugSnapshot(),
      workerRoutes: this.workerRoutes.size,
    }
  }
}

function errorToSafeShape(err: unknown): { code: string; message: string; category: string } {
  if (typeof err === 'object' && err !== null && 'code' in err && 'category' in err) {
    const e = err as { code: string; message: string; category: string }
    return { code: e.code, message: e.message, category: e.category }
  }
  return { code: 'unknown', message: err instanceof Error ? err.message : String(err), category: 'system' }
}
```

```ts
import { WorkerBroker, type WorkerBrokerConfig } from '../worker-broker'
import { MockWorker } from './mock-worker'
import type { BrokerEvent } from '@sembridge/core'

const COLLECT_PATTERNS: readonly string[] = ['*', '*.*', '*.*.*', '*.*.*.*']

export interface CollectedEvent {
  readonly topic: string
  readonly payload: unknown
  readonly source: BrokerEvent['source']
  readonly correlationId?: string
  readonly timestamp: number
}

export interface WorkerHarness {
  readonly broker: WorkerBroker
  readonly events: CollectedEvent[]
  reset(): void
  flushAsync(ms?: number): Promise<void>
}

export function createWorkerHarness(config: Partial<WorkerBrokerConfig> = {}): WorkerHarness {
  MockWorker.reset()
  const broker = new WorkerBroker({
    ...config,
    WorkerCtor: MockWorker as unknown as typeof Worker,
  })
  const events: CollectedEvent[] = []
  for (const pattern of COLLECT_PATTERNS) {
    broker.subscribe(pattern, (event) => {
      events.push({
        topic: event.topic,
        payload: event.payload,
        source: event.source,
        correlationId: (event as BrokerEvent & { correlationId?: string }).correlationId,
        timestamp: event.timestamp,
      })
    })
  }
  return {
    broker,
    events,
    reset() {
      events.length = 0
      MockWorker.reset()
    },
    async flushAsync(ms = 10) {
      await new Promise((r) => setTimeout(r, ms))
    },
  }
}
```
