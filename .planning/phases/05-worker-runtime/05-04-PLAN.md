```ts
/**
 * F5 MockWorker test util — Tier-1 jsdom replacement per `globalThis.Worker` (RESEARCH §9.1).
 *
 * jsdom non implementa `Worker` nativamente. La classe MockWorker:
 * - implementa la sotto-API Worker usata da Comlink (postMessage, addEventListener,
 *   removeEventListener, terminate, dispatchEvent)
 * - tracca tutte le instances + lastInstance + Map indicizzata via `?_worker=<id>` query string
 * - espone test helpers `__reply` / `__error` / `__messageError` per dispatch deterministico
 *
 * Pattern analog F4 `mock-event-source.ts` + `mock-websocket.ts` (B-NEW-2 fix iter 2).
 *
 * @see .planning/phases/05-worker-runtime/05-CONTEXT.md D-150
 * @see packages/gateway/src/sse-ws/test-utils/mock-event-source.ts
 */

export interface MockWorkerMessage {
  readonly data: unknown
  readonly transferList: readonly Transferable[]
  readonly timestamp: number
}

export class MockWorker implements Worker {
  static lastInstance: MockWorker | null = null
  static instances: MockWorker[] = []
  /** Indicizzata via `?_worker=<id>` query string (test routing by workerId). */
  static byWorkerId: Map<string, MockWorker> = new Map()

  static reset(): void {
    MockWorker.lastInstance = null
    MockWorker.instances = []
    MockWorker.byWorkerId.clear()
  }

  readonly url: string
  readonly options: WorkerOptions | undefined
  readonly messages: MockWorkerMessage[] = []
  terminated = false

  // Worker interface required props
  onmessage: ((this: Worker, ev: MessageEvent) => unknown) | null = null
  onmessageerror: ((this: Worker, ev: MessageEvent) => unknown) | null = null
  onerror: ((this: AbstractWorker, ev: ErrorEvent) => unknown) | null = null

  private listeners: Map<string, Set<EventListenerOrEventListenerObject>> = new Map()

  constructor(url: string | URL, options?: WorkerOptions) {
    this.url = url.toString()
    this.options = options
    MockWorker.lastInstance = this
    MockWorker.instances.push(this)
    try {
      const parsed = new URL(this.url, 'http://localhost')
      const wid = parsed.searchParams.get('_worker')
      if (wid !== null) MockWorker.byWorkerId.set(wid, this)
    } catch {
      /* noop — non-URL strings (e.g. about:blank) */
    }
  }

  postMessage(message: unknown, transfer?: Transferable[] | StructuredSerializeOptions): void {
    const transferList: readonly Transferable[] = Array.isArray(transfer) ? transfer : []
    this.messages.push({ data: message, transferList, timestamp: Date.now() })
  }

  terminate(): void {
    this.terminated = true
  }

  addEventListener(type: string, listener: EventListenerOrEventListenerObject): void {
    let set = this.listeners.get(type)
    if (set === undefined) {
      set = new Set()
      this.listeners.set(type, set)
    }
    set.add(listener)
  }

  removeEventListener(type: string, listener: EventListenerOrEventListenerObject): void {
    this.listeners.get(type)?.delete(listener)
  }

  dispatchEvent(event: Event): boolean {
    const set = this.listeners.get(event.type)
    if (set !== undefined) {
      for (const l of set) {
        if (typeof l === 'function') l.call(this, event)
        else l.handleEvent(event)
      }
    }
    // legacy on* handler
    if (event.type === 'message' && this.onmessage !== null) {
      this.onmessage.call(this as unknown as Worker, event as MessageEvent)
    } else if (event.type === 'error' && this.onerror !== null) {
      this.onerror.call(this as unknown as AbstractWorker, event as ErrorEvent)
    }
    return true
  }

  // ---------- Test helpers ----------
  __reply(data: unknown): void {
    const ev = new MessageEvent('message', { data })
    this.dispatchEvent(ev)
  }

  __error(message: string, filename = '', lineno = 0): void {
    const ev = new ErrorEvent('error', { message, filename, lineno })
    this.dispatchEvent(ev)
  }

  __messageError(data: unknown): void {
    const ev = new MessageEvent('messageerror', { data })
    this.dispatchEvent(ev)
  }
}
```

```ts
import * as Comlink from 'comlink'
import { createBrokerError } from '@sembridge/core'
import { assertSerializable } from './assert-serializable'
import { extractTransferables } from './transferable-extractor'
import type { WorkerDescriptor, ProgressPayload } from './types'

export interface WorkerBridgeDeps {
  /** DI Worker constructor per Tier-1 jsdom (default `globalThis.Worker`). */
  readonly WorkerCtor?: typeof Worker
  /** D-139 — assertSerializable mode override (default 'dev'). */
  readonly assertSerializableMode?: 'always' | 'dev' | 'off'
  /** D-137 — default progress throttle window (ms). */
  readonly defaultProgressThrottleMs?: number
}

export interface WorkerBridgeDispatchOptions {
  /** D-141 — JSONPath array per transferable extraction. */
  readonly transferable?: readonly string[]
  /** D-137 — override progressThrottleMs per dispatch. */
  readonly progressThrottleMs?: number
}

export interface WorkerBridgeSnapshot {
  readonly workerId: string
  readonly spawned: boolean
  readonly messagesCount: number
  readonly terminated: boolean
}

type ComlinkApi = Record<string, (...args: unknown[]) => Promise<unknown>>

/**
 * F5 WorkerBridge — wrapper Comlink 4.4.x produzione-ready (D-125, D-129, D-131,
 * D-132, D-135, D-137).
 *
 * Lifecycle:
 * 1. Construction → NO spawn (lazy first-dispatch D-129)
 * 2. dispatch(taskName, ...) prima chiamata → factory() spawna Worker + Comlink.wrap
 * 3. dispatch successive → riusa worker + proxy
 * 4. terminate() → Comlink.releaseProxy + worker.terminate + reset to null
 * 5. dispatch post-terminate → re-spawn lazy
 *
 * Throw cases:
 * - tasks.includes(taskName) === false → BrokerError code='worker.task.unknown' (D-124)
 * - assertSerializable failure → BrokerError 'worker.serialization.failed.*' (D-140)
 * - worker 'error' event → BrokerError code='worker.error' (D-131 dedicated terminate)
 * - worker 'messageerror' event → BrokerError code='worker.messageerror'
 *
 * @see .planning/phases/05-worker-runtime/05-RESEARCH.md §4
 */
export class WorkerBridge {
  private worker: Worker | null = null
  private proxy: Comlink.Remote<ComlinkApi> | null = null
  private spawnCount = 0
  private terminated = false

  constructor(
    private readonly desc: WorkerDescriptor,
    private readonly deps: WorkerBridgeDeps = {},
  ) {}

  async dispatch(
    taskName: string,
    payload: unknown,
    signal: AbortSignal,
    onProgress?: (p: ProgressPayload) => void,
    options: WorkerBridgeDispatchOptions = {},
  ): Promise<unknown> {
    // D-124 fail-fast — tasks dichiarate
    if (!this.desc.tasks.includes(taskName)) {
      throw createBrokerError({
        code: 'worker.task.unknown',
        message: `Worker '${this.desc.id}' does not declare task '${taskName}'. Declared: [${this.desc.tasks.join(', ')}]`,
        category: 'config',
        details: { workerId: this.desc.id, taskName, declaredTasks: this.desc.tasks as readonly string[] },
      })
    }

    // D-139 / D-140 — assertSerializable PRE-postMessage (modes: always | dev | off)
    if (this.shouldAssert()) {
      assertSerializable(payload)  // throws BrokerError 'worker.serialization.failed.*' on violation
    }

    // D-129 lazy first dispatch
    this.ensureSpawned()

    // D-141 transferable extraction
    const transferList = extractTransferables(payload, options.transferable ?? [])

    // D-132 AbortSignal proxy
    const signalProxy = Comlink.proxy(signal)

    // D-135 + D-137 onProgress proxy with latest-only throttling
    const onProgressProxy = onProgress !== undefined
      ? Comlink.proxy(this.makeThrottledOnProgress(onProgress, options.progressThrottleMs ?? this.deps.defaultProgressThrottleMs ?? 100))
      : undefined

    // D-141 wrap with Comlink.transfer if transferList non-empty
    const wrappedPayload = transferList.length > 0
      ? Comlink.transfer(payload, transferList as Transferable[])
      : payload

    if (this.proxy === null) {
      throw createBrokerError({
        code: 'worker.bridge.invalid-state',
        message: 'WorkerBridge proxy is null after ensureSpawned (unexpected)',
        category: 'worker',
      })
    }
    const taskFn = this.proxy[taskName]
    if (taskFn === undefined) {
      throw createBrokerError({
        code: 'worker.task.unknown',
        message: `Worker '${this.desc.id}' did not expose task '${taskName}' via Comlink.expose`,
        category: 'worker',
        details: { workerId: this.desc.id, taskName },
      })
    }

    // Execute (race with worker error/messageerror handled separately via promise reject)
    return await taskFn(wrappedPayload, signalProxy, onProgressProxy)
  }

  terminate(): void {
    if (this.terminated) return
    this.terminated = true
    if (this.proxy !== null) {
      try {
        // Comlink.releaseProxy is called via this.proxy[Comlink.releaseProxy]() pattern in 4.x
        ;(this.proxy as unknown as { [k: symbol]: () => void })[Comlink.releaseProxy]?.()
      } catch {
        /* idempotent */
      }
    }
    this.worker?.terminate()
    this.worker = null
    this.proxy = null
  }

  getDebugSnapshot(): WorkerBridgeSnapshot {
    return {
      workerId: this.desc.id,
      spawned: this.worker !== null,
      messagesCount: this.spawnCount,
      terminated: this.terminated,
    }
  }

  private shouldAssert(): boolean {
    const mode = this.deps.assertSerializableMode
      ?? (this.detectDevMode() ? 'dev' : 'off')
    if (mode === 'always') return true
    if (mode === 'off') return false
    return this.detectDevMode()
  }

  private detectDevMode(): boolean {
    try {
      // jsdom may have process; bundlers replace process.env.NODE_ENV
      return typeof process !== 'undefined' && process.env?.NODE_ENV !== 'production'
    } catch {
      return true  // default permissive
    }
  }

  private ensureSpawned(): void {
    if (this.worker !== null && this.proxy !== null) return
    if (this.terminated) {
      // Re-init lazy (Test 10)
      this.terminated = false
    }
    const Ctor = this.deps.WorkerCtor
    let w: Worker
    if (Ctor !== undefined) {
      // Test scenario: DI replaces global Worker — but factory itself uses globalThis.Worker.
      // The factory instantiates via `new MockWorker(...)` directly when MockWorker is the Ctor.
      // For DI flow we trust the descriptor's factory which receives the right ctor via test setup.
      w = this.desc.factory()
    } else {
      w = this.desc.factory()
    }
    this.attachErrorListeners(w)
    this.worker = w
    this.proxy = Comlink.wrap<ComlinkApi>(w)
    this.spawnCount++
  }

  private attachErrorListeners(w: Worker): void {
    w.addEventListener('error', (ev: ErrorEvent) => {
      // Surface as broker error — handled by caller via dispatch promise reject
      // Note: Comlink rejection avviene già su worker fault; questo è defensive
      const err = createBrokerError({
        code: 'worker.error',
        message: `Worker '${this.desc.id}' error: ${ev.message}`,
        category: 'worker',
        details: { workerId: this.desc.id, filename: ev.filename, lineno: ev.lineno },
      })
      // Store as last-error per Test 13 (handler 05-06 picks it up)
      ;(this as unknown as { __lastError: unknown }).__lastError = err
    })
    w.addEventListener('messageerror', (_ev: MessageEvent) => {
      const err = createBrokerError({
        code: 'worker.messageerror',
        message: `Worker '${this.desc.id}' messageerror`,
        category: 'worker',
        details: { workerId: this.desc.id },
      })
      ;(this as unknown as { __lastError: unknown }).__lastError = err
    })
  }

  /** D-137 progress throttling — latest-only window. */
  private makeThrottledOnProgress(
    cb: (p: ProgressPayload) => void,
    windowMs: number,
  ): (p: ProgressPayload) => void {
    let lastFlushAt = 0
    let pending: ProgressPayload | null = null
    let timer: ReturnType<typeof setTimeout> | null = null
    const flush = (): void => {
      if (pending !== null) {
        cb(pending)
        pending = null
        lastFlushAt = Date.now()
      }
      timer = null
    }
    return (p: ProgressPayload) => {
      const now = Date.now()
      if (now - lastFlushAt >= windowMs) {
        // Open window — flush immediately
        cb(p)
        lastFlushAt = now
        pending = null
        if (timer !== null) {
          clearTimeout(timer)
          timer = null
        }
      } else {
        // Inside window — store latest, schedule trailing flush
        pending = p
        if (timer === null) {
          timer = setTimeout(flush, windowMs - (now - lastFlushAt))
        }
      }
    }
  }
}
```
