<!-- Pattern declaration merging TS -->

```typescript
// Sintassi `declare module 'X'` in un file con almeno 1 import/export top-level.
// Il file è "module" perché esporta nulla ma ha import; il declare module aggiunge
// alla shape esistente di X.

declare module '@gluezero/core' {
  // Extend literal union (additive — non rompe F1):
  // Approach: declaration merging di literal type non è supportato direttamente.
  // Soluzione: estendere via interface merging il container.
  // PipelineStep è `type` literal — usare un namespace alternativo o trick.
  //
  // Approccio scelto: il PipelineStep di F1 È volutamente extensible.
  // Per estendere un union literal, F2 fornisce un re-export del tipo unito:
  //
  //   export type PipelineStep = '...' | F2PipelineSteps
  //
  // Il consumer importa da '@gluezero/mapper' (barrel) che ri-esporta il tipo unito.

  // Per PluginDescriptor (interface): merging interface è supportato.
  interface PluginDescriptor {
    readonly inputMap?: InputMap
    readonly outputMap?: OutputMap
    readonly canonicalSchemaId?: CanonicalSchemaId
  }

  interface BrokerConfig {
    canonicalModel?: { schemas?: CanonicalSchema[] }
    aliasRegistry?: { global?: Record<string, string>; scoped?: Record<string, Record<string, string>> }
    transforms?: Record<string, TransformFn>
  }
}
```

NOTA tecnica: `PipelineStep` è un type alias literal union, NON un'interface. TS NON supporta declaration merging di type alias. Soluzione concreta:
- `@gluezero/mapper` ri-esporta `PipelineStep` come `import('@gluezero/core').PipelineStep | F2PipelineSteps` — il consumer di `@gluezero/mapper` importa il tipo dal mapper barrel (NON dal core). Il tipo finale è il super-set.
- Documentato in JSDoc di entrambi i barrel.

```ts
// augment.ts — TS declaration merging per estendere @gluezero/core con i tipi F2.
// (D-49, D-50, D-56, D-57)
//
// Vincolo D-49: NESSUNA modifica a packages/core/src/. Solo declaration merging dal mapper.
//
// Cosa estende:
//   - PluginDescriptor — aggiunge inputMap, outputMap, canonicalSchemaId (D-57)
//   - BrokerConfig — sostituisce `unknown` placeholder con tipi specifici (D-56)
//
// Cosa NON estende qui:
//   - PipelineStep (type alias literal): TS non supporta declaration merging di type alias.
//     Strategia: il barrel @gluezero/mapper ri-esporta `PipelineStep` come unione di
//     `core.PipelineStep | F2PipelineStep`. I consumer importano dal mapper.
//
// Side-effect import: `packages/mapper/src/index.ts` importa questo file per side-effect.
// `package.json` ha `sideEffects: ["./dist/augment.js"]` per evitare tree-shaking accidentale.

import type { CanonicalSchema, CanonicalSchemaId, InputMap, OutputMap, TransformFn } from './types'

declare module '@gluezero/core' {
  /**
   * F2 augmentation: aggiunge inputMap, outputMap, canonicalSchemaId al PluginDescriptor.
   * (D-57 — chiude i placeholder F1 in packages/core/src/types/plugin.ts:47-51).
   */
  interface PluginDescriptor {
    readonly inputMap?: InputMap
    readonly outputMap?: OutputMap
    readonly canonicalSchemaId?: CanonicalSchemaId
  }

  /**
   * F2 augmentation: sostituisce `unknown` placeholder con tipi specifici per le sezioni
   * canonicalModel, aliasRegistry, transforms del BrokerConfig.
   * (D-56 — chiude i placeholder F1 in packages/core/src/types/config.ts:53-58).
   */
  interface BrokerConfig {
    canonicalModel?: {
      readonly schemas?: readonly CanonicalSchema[]
    }
    aliasRegistry?: {
      readonly global?: Readonly<Record<string, string>>
      readonly scoped?: Readonly<Record<string, Readonly<Record<string, string>>>>
    }
    transforms?: Readonly<Record<string, TransformFn>>
  }
}

// File ha bisogno di almeno un import/export top-level per essere considerato un module
// (vs ambient declarations). Il `import type` sopra soddisfa il requirement.
// Dummy export per double-safety + verificare side-effect import non venga tree-shaken.
export const __augmentLoaded: true = true
```
