```ts
// MapperBroker — composition wrapper di Broker (F1) con MapperEngine + Inspector (F2).
// Riferimento decisioni: D-31 (API), D-49 (no modify bus.ts), D-50 (5 step F2),
// D-51 (delivery loop applica inputMap), D-58 (mapping.error), D-26 ext (cascade).
//
// Strategia: composizione (NON subclass). Internamente compone:
//   - inner: Broker (F1) — delegato per pub/sub/lifecycle base
//   - canonical: CanonicalRegistry (Wave 3)
//   - alias: AliasRegistry (Wave 3)
//   - transforms: TransformPipeline (Wave 3)
//   - mapper: MapperEngine (Wave 4)
//   - inspector: MappingInspector (Wave 5)
//
// Surface pubblica F2 (wrapped F1 + F2 new):
//   publish<T>(topic, payload, options?): applica outputMap + canonical validation, poi inner.publish
//   subscribe(pattern, handler, options?): wrappa handler con applyInputMap del consumer
//   registerPlugin(desc): chiama inner + compileMappings + tracking ownership
//   unregisterPlugin(id): cascade — inner + alias.unregisterScopedAll + transforms.unregisterByOwner + mapper.unregisterPluginMappings
//   registerCanonicalSchema(schema): boolean (D-31)
//   registerTransform(name, fn, options?): void (D-31)
//   registerAlias(local, canonical, options?: { scope?: 'global' | string }): boolean (D-31)
//   getMappingInspector(): MappingInspector
//   getDebugSnapshot(): BrokerDebugSnapshot esteso con sezione `mappings` (D-48)
//
// Threat coverage:
// - T-02-10-01 (DoS — mapping error storm): inspector ring buffer bounded.
// - T-02-10-02 (Tampering — descriptor mutation post-register): cycle detection register-time.
// - T-02-10-03 (Repudiation — cascade incompleta): try/catch swallow per ogni step cascade.

import {
  Broker, createBrokerError, isBrokerError, silentLogger,
} from '@gluezero/core'
import type {
  BrokerConfig, BrokerEvent, BrokerError, BrokerLogger,
  PluginDescriptor, Subscription, SubscribeOptions,
} from '@gluezero/core'
import { CanonicalRegistry } from './canonical-registry'
import { AliasRegistry } from './alias-registry'
import { TransformPipeline } from './transform-pipeline'
import { MapperEngine, type MapperPluginDescriptor } from './mapper-engine'
import { MappingInspector } from './inspector'
import { valibotAdapter } from './valibot-adapter'
import type { CanonicalSchema, CanonicalSchemaId, TransformFn } from './types'

export interface RegisterAliasOptions {
  /** `'global'` (default) o `pluginId` per scoped. */
  readonly scope?: 'global' | string
}

export interface MapperBrokerDebugSnapshot {
  // F1 fields propagati
  readonly topics: string[]
  readonly subscriberCount: Record<string, number>
  readonly pluginIds: string[]
  readonly pendingAsyncDelivery: number
  readonly logLevel: string
  readonly pipelineSteps: string[]
  // F2 extension (D-48)
  readonly mappings: {
    readonly canonicalSchemas: number
    readonly registeredAliases: number
    readonly registeredTransforms: number
    readonly lastMappingErrors: BrokerError[]
  }
}

export class MapperBroker {
  private readonly inner: Broker
  private readonly canonicalRegistry: CanonicalRegistry
  private readonly aliasRegistry: AliasRegistry
  private readonly transformPipeline: TransformPipeline
  private readonly mapper: MapperEngine
  private readonly inspector: MappingInspector
  private readonly logger: BrokerLogger

  /** Track ownership: pluginId → { canonicalSchemaIds, aliasScoped, transformNames } */
  private readonly ownership = new Map<
    string,
    {
      canonicalSchemaIds: Set<string>
      transformNames: Set<string>
    }
  >()

  constructor(config: BrokerConfig = {}) {
    this.logger = config.runtime?.logger ?? silentLogger
    this.canonicalRegistry = new CanonicalRegistry()
    this.aliasRegistry = new AliasRegistry()
    this.transformPipeline = new TransformPipeline()
    this.mapper = new MapperEngine({
      canonicalRegistry: this.canonicalRegistry,
      aliasRegistry: this.aliasRegistry,
      transformPipeline: this.transformPipeline,
      validator: valibotAdapter,
      logger: this.logger,
    })
    this.inspector = new MappingInspector({
      canonicalRegistry: this.canonicalRegistry,
      aliasRegistry: this.aliasRegistry,
      transformPipeline: this.transformPipeline,
    })
    this.inner = new Broker(config)

    // Bootstrap from config (D-56 wired)
    if (config.canonicalModel?.schemas) {
      for (const schema of config.canonicalModel.schemas) {
        this.canonicalRegistry.register(schema)
      }
    }
    if (config.aliasRegistry?.global) {
      for (const [local, canonical] of Object.entries(config.aliasRegistry.global)) {
        this.aliasRegistry.registerGlobal(local, canonical)
      }
    }
    if (config.transforms) {
      for (const [name, fn] of Object.entries(config.transforms)) {
        this.transformPipeline.register(name, fn as TransformFn)
      }
    }
  }

  // === F1 surface delegated + wrapped ===

  publish<T>(
    topic: string,
    payload: T,
    options: { readonly source?: { type: string; id: string }; readonly [k: string]: unknown } = {},
  ): void {
    // Step 4: source resolution
    const sourcePluginId = options.source?.id
    let canonicalPayload: unknown = payload
    if (sourcePluginId !== undefined) {
      try {
        // Step 5: applyOutputMap
        canonicalPayload = this.mapper.applyOutputMap(sourcePluginId, payload)
        // Step 6: canonical validation (deferred to per-schema integration)
      } catch (err) {
        if (isBrokerError(err)) {
          this.inspector.recordError(err)
          // D-58: publish mapping.error
          this.inner.publish(
            'mapping.error',
            { error: err, sourceEvent: topic, step: 'event.mapped.canonical' },
            { source: { type: 'system', id: 'mapper' }, deliveryMode: 'async' },
          )
          // D-59: NON consegnare al subscriber (fail-fast)
          return
        }
        throw err
      }
    }
    this.inner.publish(topic, canonicalPayload, options)
  }

  subscribe(
    pattern: string,
    handler: (event: BrokerEvent) => void | Promise<void>,
    options: SubscribeOptions = {},
  ): Subscription {
    // D-51: applica inputMap del consumer (passi 11+12) PRIMA di consegnare al handler reale.
    // ownerId viene settato automaticamente da createPluginScopedBroker (F1 plan 08) quando
    // il subscribe avviene da dentro un plugin hook. Per chiamate dirette (test, top-level
    // consumer) ownerId è undefined → handler riceve payload canonico puro (passthrough).
    const consumerPluginId = options.ownerId
    if (consumerPluginId === undefined) {
      return this.inner.subscribe(pattern, handler, options)
    }
    const wrappedHandler = (event: BrokerEvent): void | Promise<void> => {
      const tapStart = startStep()
      try {
        // Passo 11: applyInputMap consumer-side (canonical → consumer locale)
        const mappedPayload = this.mapper.applyInputMap(consumerPluginId, event.payload)
        const mappedEvent: BrokerEvent = { ...event, payload: mappedPayload }
        safeTapStep(this.tap, 'event.mapped.consumer', tapStart('event.mapped.consumer', event.id, event.topic, { payloadBefore: event.payload, payloadAfter: mappedPayload }))
        // Passo 12: final validation per consumer (D-39 + VAL-04). In F2 V1 questo è
        // structural-only (FieldDescriptor presence); typed validation runtime via
        // Valibot deferred a V1.x.
        safeTapStep(this.tap, 'event.final.validated', tapStart('event.final.validated', event.id, event.topic))
        return handler(mappedEvent)
      } catch (err) {
        if (isBrokerError(err)) {
          this.inspector.recordError(err)
          // D-58: publish mapping.error; D-59: NO <topic>.failed
          this.inner.publish('mapping.error', { error: err, sourceEvent: event.id, step: 'event.mapped.consumer' as const }, { source: { type: 'system', id: 'mapper' } })
          // D-26: NON consegnare a questo consumer (gli altri matched ricevono comunque)
          return
        }
        throw err
      }
    }
    return this.inner.subscribe(pattern, wrappedHandler, options)
  }

  async registerPlugin(descriptor: PluginDescriptor): Promise<void> {
    // Pre-register: cycle detection IMMEDIATAMENTE (D-35)
    const mp = descriptor as MapperPluginDescriptor
    if (mp.outputMap || mp.inputMap) {
      this.mapper.compileMappings(mp) // throws on cycle
    }
    await this.inner.registerPlugin(descriptor)
    this.ownership.set(descriptor.id, {
      canonicalSchemaIds: new Set(),
      transformNames: new Set(),
    })
  }

  async unregisterPlugin(id: string): Promise<void> {
    await this.inner.unregisterPlugin(id)
    // Cascade D-26 ext F2 (LIFE-02 ext)
    this.aliasRegistry.unregisterScopedAll(id)
    this.transformPipeline.unregisterByOwner(id)
    this.mapper.unregisterPluginMappings(id)
    const own = this.ownership.get(id)
    if (own) {
      for (const schemaId of own.canonicalSchemaIds) {
        this.canonicalRegistry.unregister(schemaId as CanonicalSchemaId)
      }
      this.ownership.delete(id)
    }
  }

  getTopicRegistry(): readonly string[] {
    return this.inner.getTopicRegistry()
  }

  setLogger(logger: BrokerLogger): void {
    this.inner.setLogger(logger)
  }

  enableDebug(): void {
    this.inner.enableDebug()
  }

  disableDebug(): void {
    this.inner.disableDebug()
  }

  getDebugSnapshot(): MapperBrokerDebugSnapshot {
    const inner = this.inner.getDebugSnapshot()
    const insp = this.inspector.getSnapshot()
    return {
      topics: inner.topics,
      subscriberCount: inner.subscriberCount,
      pluginIds: inner.pluginIds,
      pendingAsyncDelivery: inner.pendingAsyncDelivery,
      logLevel: inner.logLevel,
      pipelineSteps: [...inner.pipelineSteps],
      mappings: {
        canonicalSchemas: insp.canonicalSchemas,
        registeredAliases: insp.registeredAliases,
        registeredTransforms: insp.registeredTransforms,
        lastMappingErrors: insp.lastMappingErrors,
      },
    }
  }

  // === F2 new API surface (D-31) ===

  registerCanonicalSchema(schema: CanonicalSchema, options?: { ownerId?: string }): boolean {
    const ok = this.canonicalRegistry.register(schema)
    if (ok && options?.ownerId !== undefined) {
      const own = this.ownership.get(options.ownerId)
      if (own) own.canonicalSchemaIds.add(schema.id)
    }
    return ok
  }

  registerTransform(name: string, fn: TransformFn, options?: { ownerId?: string }): void {
    this.transformPipeline.register(name, fn, options ?? {})
    if (options?.ownerId !== undefined) {
      const own = this.ownership.get(options.ownerId)
      if (own) own.transformNames.add(name)
    }
  }

  registerAlias(localField: string, canonicalField: string, options: RegisterAliasOptions = {}): boolean {
    const scope = options.scope ?? 'global'
    if (scope === 'global') {
      return this.aliasRegistry.registerGlobal(localField, canonicalField)
    }
    return this.aliasRegistry.registerScoped(scope, localField, canonicalField)
  }

  getMappingInspector(): MappingInspector {
    return this.inspector
  }
}

// public-factory.ts — createMapperBroker(config) per @gluezero/mapper.
// Estende createBroker di F1 (D-19, D-30) con validation Valibot delle sezioni F2
// (canonicalModel, aliasRegistry, transforms — D-56).

import * as v from 'valibot'
import type { BrokerConfig } from '@gluezero/core'
import { MapperBroker } from './broker-mapper-wrapper'

const FieldDescriptorSchema = v.object({
  type: v.picklist(['string', 'number', 'boolean', 'object', 'array', 'any']),
  required: v.optional(v.boolean()),
  default: v.optional(v.unknown()),
  onFailure: v.optional(v.picklist(['block', 'skip', 'fallback'])),
  description: v.optional(v.string()),
})

const CanonicalSchemaSchema = v.object({
  id: v.string(),
  requires: v.optional(v.array(v.string())),
  fields: v.record(v.string(), FieldDescriptorSchema),
  description: v.optional(v.string()),
})

const MapperBrokerConfigSchema = v.object({
  runtime: v.optional(v.unknown()),
  debug: v.optional(v.unknown()),
  canonicalModel: v.optional(
    v.object({
      schemas: v.optional(v.array(CanonicalSchemaSchema)),
    }),
  ),
  aliasRegistry: v.optional(
    v.object({
      global: v.optional(v.record(v.string(), v.string())),
      scoped: v.optional(v.record(v.string(), v.record(v.string(), v.string()))),
    }),
  ),
  transforms: v.optional(v.record(v.string(), v.function())),
  routes: v.optional(v.unknown()),
  transport: v.optional(v.unknown()),
  workers: v.optional(v.unknown()),
  topicSchemas: v.optional(v.unknown()),
  cache: v.optional(v.unknown()),
})

/**
 * Creates a new {@link MapperBroker} instance with the given configuration.
 *
 * Estende {@link createBroker} (F1) con validation delle sezioni F2:
 * `canonicalModel`, `aliasRegistry`, `transforms` (D-56).
 *
 * No singleton (D-30): ogni call ritorna istanza indipendente.
 *
 * @param config - Optional broker configuration.
 * @returns A fresh {@link MapperBroker} instance.
 * @throws {Error} `Invalid MapperBrokerConfig: ...` se config validation fallisce.
 *
 * @example
 * ```ts
 * import { createMapperBroker } from '@gluezero/mapper'
 *
 * const broker = createMapperBroker({
 *   runtime: { logLevel: 'info' },
 *   canonicalModel: {
 *     schemas: [{ id: 'weather', fields: { location: { type: 'string', required: true } } }],
 *   },
 *   transforms: {
 *     parseItalianDate: (input) => { ... },
 *   },
 * })
 * ```
 */
export function createMapperBroker(config: BrokerConfig = {}): MapperBroker {
  const parsed = v.safeParse(MapperBrokerConfigSchema, config)
  if (!parsed.success) {
    const messages = parsed.issues.map((i) => i.message).join('; ')
    throw new Error(`Invalid MapperBrokerConfig: ${messages}`)
  }
  return new MapperBroker(config)
}

export { MapperBroker }
```
