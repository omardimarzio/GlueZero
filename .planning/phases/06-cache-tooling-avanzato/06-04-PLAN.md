```typescript
/**
 * F6 MultiplexTap — aggregator chain N EventTap con error isolation try/catch
 * isolato per tap (D-159 + D-20 carryover safeTapStep F1).
 *
 * Failure di un tap NON ferma downstream tap né blocca pipeline. Pattern Adapter
 * classico — la pipeline F1 vede UN solo tap (single value `runtime.tap`), ma
 * quel tap delega a N tap interni F6.
 *
 * @see packages/core/src/core/bus.ts safeTapStep (analog F1 D-20)
 * @see RESEARCH §5.2 estensione F6 tap registry chain
 *
 * @example
 * const aggregated = createMultiplexTap([eventInspector.tap, metricsCollector.tap, customAnalyticsTap])
 * // Wired al RouterBroker come single tap:
 * // new RouterBroker({ ...config, runtime: { tap: aggregated } })
 */

import type { EventTap, PipelineSnapshot, PipelineStep } from '@sembridge/core'

export function createMultiplexTap(taps: readonly EventTap[]): EventTap {
  return {
    onPipelineStep(step: PipelineStep, snapshot: PipelineSnapshot): void {
      for (const tap of taps) {
        try {
          tap.onPipelineStep(step, snapshot)
        } catch {
          // swallow — pattern F1 safeTapStep D-20 carryover
          // Failure di un tap NON ferma downstream taps né blocca pipeline (D-159)
        }
      }
    },
  }
}
```

```typescript
/**
 * F6 TapRegistry — runtime tap management + auto-wrap F1 single-tap backward-compat (D-159).
 *
 * Pattern Map registry (analog F5 worker-registry.ts:104-110).
 *
 * @see RESEARCH §5.2 auto-wrap F1 backward-compat
 */

import type { EventTap } from '@sembridge/core'
import { createMultiplexTap } from './multiplex-tap'

export interface TapHandle {
  readonly id: string
}

export interface TapRegistry {
  add(tap: EventTap): TapHandle
  remove(handle: TapHandle): boolean
  list(): readonly EventTap[]
  getMultiplexed(): EventTap
}

export function createTapRegistry(): TapRegistry {
  const taps = new Map<string, EventTap>()
  let nextId = 0
  return {
    add(tap) {
      const id = `tap-${++nextId}`
      taps.set(id, tap)
      return { id }
    },
    remove(handle) {
      return taps.delete(handle.id)
    },
    list() {
      return Array.from(taps.values())
    },
    getMultiplexed() {
      return createMultiplexTap(Array.from(taps.values()))
    },
  }
}

/**
 * Auto-wrap F1 single-tap legacy per backward-compat zero breaking (D-159).
 *
 * @param config Subset di BrokerConfig con `runtime.tap` (legacy F1) + `taps` (F6 array)
 * @returns Array consolidato — ordering: F6 array first, F1 legacy last
 */
export function wrapLegacyTap(config: {
  readonly runtime?: { readonly tap?: EventTap }
  readonly taps?: readonly EventTap[]
}): readonly EventTap[] {
  const userTaps = config.taps ?? []
  if (config.runtime?.tap) {
    return [...userTaps, config.runtime.tap]
  }
  return userTaps
}
```
